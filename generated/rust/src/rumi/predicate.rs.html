1//! Predicate — Boolean expressions over `DataInputs`
2//!
3//! Predicates combine [`DataInput`] and [`InputMatcher`] to create boolean
4//! conditions that can be composed with AND/OR/NOT.
5
6use crate::{DataInput, InputMatcher, MatchingData};
7use std::fmt::Debug;
8
9/// A single predicate: combines a [`DataInput`] with an [`InputMatcher`].
10///
11/// This is where domain-specific (`DataInput`) meets domain-agnostic (`InputMatcher`).
12/// The `SinglePredicate` extracts data from the context using the input, then
13/// passes it to the matcher for evaluation.
14///
15/// # INV (Dijkstra): None → false
16///
17/// If the `DataInput` returns [`MatchingData::None`], the predicate evaluates to `false`.
18/// This is a critical invariant that simplifies reasoning about matcher behavior.
19///
20/// # Example
21///
22/// ```ignore
23/// let predicate = SinglePredicate::new(
24///     Box::new(HeaderInput::new("content-type")),
25///     Box::new(ContainsMatcher::new("json")),
26/// );
27/// let result = predicate.evaluate(&request);
28/// ```
29pub struct SinglePredicate<Ctx> {
30    input: Box<dyn DataInput<Ctx>>,
31    matcher: Box<dyn InputMatcher>,
32}
33
34impl<Ctx> SinglePredicate<Ctx> {
35    /// Create a new single predicate from a `DataInput` and `InputMatcher`.
36    #[must_use]
37    pub fn new(input: Box<dyn DataInput<Ctx>>, matcher: Box<dyn InputMatcher>) -> Self {
38        Self { input, matcher }
39    }
40
41    /// Get a reference to the input.
42    #[must_use]
43    pub fn input(&self) -> &dyn DataInput<Ctx> {
44        &*self.input
45    }
46
47    /// Get a reference to the matcher.
48    #[must_use]
49    pub fn matcher(&self) -> &dyn InputMatcher {
50        &*self.matcher
51    }
52
53    /// Evaluate this predicate against the given context.
54    ///
55    /// # Returns
56    ///
57    /// - `true` if the input produces data and the matcher matches
58    /// - `false` if the input returns `None` (INV: None → false)
59    /// - `false` if the matcher doesn't match
60    pub fn evaluate(&self, ctx: &Ctx) -> bool {
61        let data = self.input.get(ctx);
62        match data {
63            MatchingData::None => false, // INV: None → false
64            _ => self.matcher.matches(&data),
65        }
66    }
67}
68
69impl<Ctx> Debug for SinglePredicate<Ctx> {
70    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
71        f.debug_struct("SinglePredicate")
72            .field("input", &self.input)
73            .field("matcher", &self.matcher)
74            .finish()
75    }
76}

82/// Composite predicate with boolean logic.
83///
84/// Predicates can be composed using AND, OR, and NOT operations.
85/// Evaluation uses short-circuit semantics for efficiency.
86///
87/// # Variants
88///
89/// - `Single` — A single predicate
90/// - `And` — All predicates must match (short-circuit on first false)
91/// - `Or` — Any predicate must match (short-circuit on first true)
92/// - `Not` — Inverts the result of the inner predicate
93///
94/// # Example
95///
96/// ```ignore
97/// // Match requests where content-type contains "json" AND method is "POST"
98/// let predicate = Predicate::And(vec![
99///     Predicate::Single(content_type_json),
100///     Predicate::Single(method_post),
101/// ]);
102/// ```
103pub enum Predicate<Ctx> {
104    /// A single predicate.
105    Single(SinglePredicate<Ctx>),
106
107    /// All predicates must match (logical AND).
108    /// Short-circuits on the first `false`.
109    And(Vec<Predicate<Ctx>>),
110
111    /// Any predicate must match (logical OR).
112    /// Short-circuits on the first `true`.
113    Or(Vec<Predicate<Ctx>>),
114
115    /// Inverts the result of the inner predicate (logical NOT).
116    Not(Box<Predicate<Ctx>>),
117}
118
119impl<Ctx> Predicate<Ctx> {
120    /// Evaluate this predicate against the given context.
121    ///
122    /// # Safety Consideration (Taleb)
123    ///
124    /// This uses recursion. Production code should use iterative evaluation
125    /// with an explicit stack to protect against stack overflow from deeply
126    /// nested predicates. The depth limit (max 32) should be enforced at
127    /// config load time, not evaluation time.
128    pub fn evaluate(&self, ctx: &Ctx) -> bool {
129        match self {
130            Predicate::Single(p) => p.evaluate(ctx),
131            Predicate::And(predicates) => predicates.iter().all(|p| p.evaluate(ctx)),
132            Predicate::Or(predicates) => predicates.iter().any(|p| p.evaluate(ctx)),
133            Predicate::Not(p) => !p.evaluate(ctx),
134        }
135    }
136
137    /// Returns `true` if this is a `Single` predicate.
138    #[must_use]
139    pub fn is_single(&self) -> bool {
140        matches!(self, Predicate::Single(_))
141    }
142
143    /// Returns `true` if this is an `And` predicate.
144    #[must_use]
145    pub fn is_and(&self) -> bool {
146        matches!(self, Predicate::And(_))
147    }
148
149    /// Returns `true` if this is an `Or` predicate.
150    #[must_use]
151    pub fn is_or(&self) -> bool {
152        matches!(self, Predicate::Or(_))
153    }
154
155    /// Returns `true` if this is a `Not` predicate.
156    #[must_use]
157    pub fn is_not(&self) -> bool {
158        matches!(self, Predicate::Not(_))
159    }
160
161    /// Calculate the depth of this predicate tree.
162    ///
163    /// Used for depth limit validation at config time.
164    #[must_use]
165    pub fn depth(&self) -> usize {
166        match self {
167            Predicate::Single(_) => 1,
168            Predicate::And(ps) | Predicate::Or(ps) => {
169                1 + ps.iter().map(Predicate::depth).max().unwrap_or(0)
170            }
171            Predicate::Not(p) => 1 + p.depth(),
172        }
173    }
174}
175
176impl<Ctx> Debug for Predicate<Ctx> {
177    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
178        match self {
179            Predicate::Single(p) => f.debug_tuple("Single").field(p).finish(),
180            Predicate::And(ps) => f.debug_tuple("And").field(&ps.len()).finish(),
181            Predicate::Or(ps) => f.debug_tuple("Or").field(&ps.len()).finish(),
182            Predicate::Not(_) => f.debug_tuple("Not").finish(),
183        }
184    }
185}

219    #[test]
220    fn test_single_predicate_matches() {
221        let ctx = TestContext {
222            value: "hello".to_string(),

239    fn test_predicate_and() {
240        let ctx = TestContext {
241            value: "hello world".to_string(),
242        };
243
244        let pred = Predicate::And(vec![
245            Predicate::Single(SinglePredicate::new(
246                Box::new(ValueInput),
247                Box::new(ContainsMatcher::new("hello")),

249            Predicate::Single(SinglePredicate::new(
250                Box::new(ValueInput),
251                Box::new(ContainsMatcher::new("world")),
252            )),
253        ]);

255        assert!(pred.evaluate(&ctx));

259            Predicate::Single(SinglePredicate::new(

264                Box::new(ValueInput),
265                Box::new(ContainsMatcher::new("foo")),
266            )),

308        let single = Predicate::Single::<TestContext>(SinglePredicate::new(

312        assert_eq!(single.depth(), 1);
