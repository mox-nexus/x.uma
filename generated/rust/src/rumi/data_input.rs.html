1//! `DataInput` — Domain-specific data extraction
2//!
3//! The `DataInput` trait extracts data from a domain-specific context (e.g., HTTP request,
4//! Claude hook context) and returns type-erased `MatchingData`.
5//!
6//! This is generic over the context type `Ctx`, but returns domain-agnostic `MatchingData`,
7//! enabling `InputMatchers` to be shared across different domains.
8
9use crate::MatchingData;
10use std::fmt::Debug;
11
12/// Extracts data from a domain-specific context.
13///
14/// `DataInput` is the bridge between domain-specific contexts (like HTTP requests)
15/// and domain-agnostic matchers. It extracts relevant data from the context and
16/// returns it as type-erased [`MatchingData`].
17///
18/// # Type Parameters
19///
20/// - `Ctx`: The context type this input operates on (e.g., `HttpRequest`, `HookContext`)
21///
22/// # Thread Safety
23///
24/// All implementations must be `Send + Sync` to support concurrent evaluation
25/// and FFI use cases.
26///
27/// # Example
28///
29/// ```ignore
30/// use rumi::{DataInput, MatchingData};
31///
32/// #[derive(Debug)]
33/// struct HeaderInput { name: String }
34///
35/// impl DataInput<HttpRequest> for HeaderInput {
36///     fn get(&self, ctx: &HttpRequest) -> MatchingData {
37///         ctx.headers.get(&self.name)
38///             .map(|v| MatchingData::String(v.clone()))
39///             .unwrap_or(MatchingData::None)
40///     }
41/// }
42/// ```
43pub trait DataInput<Ctx>: Send + Sync + Debug {
44    /// Extract data from the given context.
45    ///
46    /// Returns [`MatchingData::None`] if the requested data is not present.
47    /// This is important for the INV (Dijkstra): None → predicate evaluates to false.
48    fn get(&self, ctx: &Ctx) -> MatchingData;
49
50    /// Returns a static string describing the type of data this input produces.
51    ///
52    /// Used for config-time validation to ensure DataInput/InputMatcher compatibility.
53    /// Default is `"string"` since most inputs produce string data.
54    fn data_type(&self) -> &'static str {
55        "string"
56    }
57}
58
59// Blanket implementation for boxed DataInputs
60impl<Ctx> DataInput<Ctx> for Box<dyn DataInput<Ctx>> {
61    fn get(&self, ctx: &Ctx) -> MatchingData {
62        (**self).get(ctx)
63    }
64
65    fn data_type(&self) -> &'static str {
66        (**self).data_type()
67    }
68}

70#[cfg(test)]
71mod tests {
72    use super::*;
73
74    #[derive(Debug)]
75    struct TestContext {
76        value: String,
77    }
78
79    #[derive(Debug)]
80    struct ValueInput;
81
82    impl DataInput<TestContext> for ValueInput {
83        fn get(&self, ctx: &TestContext) -> MatchingData {
84            MatchingData::String(ctx.value.clone())
85        }
86    }
87
88    #[test]
89    fn test_data_input_basic() {
90        let ctx = TestContext {
91            value: "hello".to_string(),
92        };
93        let input = ValueInput;
94        let data = input.get(&ctx);
95        assert_eq!(data.as_str(), Some("hello"));
96    }
97
98    #[test]
99    fn test_data_input_is_send_sync() {
100        fn assert_send_sync<T: Send + Sync>() {}
101        assert_send_sync::<Box<dyn DataInput<TestContext>>>();
102    }
