1//! Matcher — Top-level matcher with first-match-wins semantics
2//!
3//! The `Matcher` is the entry point for evaluation. It contains a list of
4//! field matchers and evaluates them in order, returning the first match.
5
6use crate::{
7    EvalStep, EvalTrace, FieldMatcher, MatcherError, OnMatch, OnMatchTrace, Predicate, MAX_DEPTH,
8};
9use std::fmt::Debug;
10use std::marker::PhantomData;
11
12/// Top-level matcher with first-match-wins semantics.
13///
14/// A Matcher contains:
15/// - A list of field matchers (predicate + action pairs)
16/// - An optional `on_no_match` fallback
17///
18/// Evaluation iterates through the field matchers in order and returns
19/// the action from the first matching predicate. If no predicate matches,
20/// the `on_no_match` action is returned (if present).
21///
22/// # Type Parameters
23///
24/// - `Ctx`: The context type to match against
25/// - `A`: The action type (must be `Clone + Send + Sync + 'static`)
26///
27/// # INV (Dijkstra): First-match-wins
28///
29/// Field matchers are evaluated in order. The first matching predicate
30/// terminates evaluation, even if later predicates would also match.
31///
32/// # xDS Semantics: Nested Matcher Failure Propagates
33///
34/// When an `OnMatch` contains a nested matcher and that nested matcher
35/// returns no match, the ENTIRE `OnMatch` fails — there is no fallback
36/// to a sibling action (because `OnMatch` is exclusive per xDS proto).
37///
38/// # Example
39///
40/// ```ignore
41/// let matcher = Matcher::new(
42///     vec![
43///         FieldMatcher::new(api_path_predicate, OnMatch::action("api".to_string())),
44///         FieldMatcher::new(static_path_predicate, OnMatch::action("static".to_string())),
45///     ],
46///     Some(OnMatch::action("default".to_string())),
47/// );
48///
49/// let action = matcher.evaluate(&request);
50/// ```
51pub struct Matcher<Ctx, A: Clone + Send + Sync + 'static> {
52    /// The list of field matchers to evaluate.
53    pub matcher_list: Vec<FieldMatcher<Ctx, A>>,
54
55    /// Fallback when no field matcher matches.
56    /// Note: per xDS, this is at the Matcher level, not per-OnMatch.
57    pub on_no_match: Option<OnMatch<Ctx, A>>,
58
59    _phantom: PhantomData<Ctx>,
60}
61
62impl<Ctx, A: Clone + Send + Sync + 'static> Matcher<Ctx, A> {
63    /// Create a new matcher.
64    pub fn new(
65        matcher_list: Vec<FieldMatcher<Ctx, A>>,
66        on_no_match: Option<OnMatch<Ctx, A>>,
67    ) -> Self {
68        Self {
69            matcher_list,
70            on_no_match,
71            _phantom: PhantomData,
72        }
73    }
74
75    /// Create a matcher from a single predicate with action and optional fallback.
76    ///
77    /// This is the common pattern for domain compilers: a predicate tree
78    /// (built with [`Predicate::from_all`] / [`Predicate::from_any`])
79    /// paired with a single action and optional fallback.
80    ///
81    /// # Example
82    ///
83    /// ```ignore
84    /// let matcher = Matcher::from_predicate(
85    ///     Predicate::from_any(rule_predicates, catch_all),
86    ///     "allow",
87    ///     Some("deny"),
88    /// );
89    /// ```
90    pub fn from_predicate(predicate: Predicate<Ctx>, action: A, on_no_match: Option<A>) -> Self {
91        Self::new(
92            vec![FieldMatcher::new(predicate, OnMatch::Action(action))],
93            on_no_match.map(OnMatch::Action),
94        )
95    }
96
97    /// Create an empty matcher (no field matchers, no fallback).
98    #[must_use]
99    pub fn empty() -> Self {
100        Self {
101            matcher_list: Vec::new(),
102            on_no_match: None,
103            _phantom: PhantomData,
104        }
105    }
106
107    /// Evaluate this matcher against the given context.
108    ///
109    /// Returns the action from the first matching field matcher,
110    /// or the `on_no_match` action if nothing matches.
111    ///
112    /// # First-match-wins semantics (INV)
113    ///
114    /// Field matchers are evaluated in order. Evaluation stops at the
115    /// first matching predicate.
116    ///
117    /// # xDS Nested Matcher Semantics
118    ///
119    /// If a field matcher's `OnMatch` contains a nested matcher:
120    /// - Evaluation continues into the nested matcher
121    /// - If the nested matcher returns Some(action), that's returned
122    /// - If the nested matcher returns None, this field matcher is considered
123    ///   NOT matched, and we continue to the next field matcher
124    ///
125    /// This matches xDS semantics where nested matcher failure propagates up.
126    pub fn evaluate(&self, ctx: &Ctx) -> Option<A> {
127        // First-match-wins: iterate through matchers, return first match
128        for field_matcher in &self.matcher_list {
129            if field_matcher.matches(ctx) {
130                // OnMatch is now an enum: either Action or Matcher
131                match &field_matcher.on_match {
132                    OnMatch::Action(action) => return Some(action.clone()),
133                    OnMatch::Matcher(nested) => {
134                        // xDS semantics: nested matcher failure propagates
135                        // If nested returns None, continue to next field_matcher
136                        if let Some(action) = nested.evaluate(ctx) {
137                            return Some(action);
138                        }
139                        // Nested returned None → this field_matcher is NOT a match
140                        // Continue to next field_matcher
141                    }
142                }
143            }
144        }
145
146        // No match: return on_no_match action if present
147        self.on_no_match.as_ref().and_then(|om| match om {
148            OnMatch::Action(a) => Some(a.clone()),
149            OnMatch::Matcher(nested) => nested.evaluate(ctx),
150        })
151    }
152
153    /// Evaluate with full trace for debugging.
154    ///
155    /// Returns the same result as [`evaluate()`](Self::evaluate) plus
156    /// the full evaluation path: which field matchers were checked, which
157    /// predicates fired, and whether the fallback was used.
158    ///
159    /// # INV: `result` == `evaluate()` result
160    ///
161    /// The returned [`EvalTrace::result`] always equals what `evaluate()`
162    /// would return for the same context.
163    #[must_use]
164    pub fn evaluate_with_trace(&self, ctx: &Ctx) -> EvalTrace<A> {
165        let mut steps = Vec::new();
166        let mut result = None;
167
168        for (index, field_matcher) in self.matcher_list.iter().enumerate() {
169            let predicate_trace = field_matcher.predicate.evaluate_with_trace(ctx);
170            let pred_matched = predicate_trace.matched();
171
172            if pred_matched {
173                let on_match = match &field_matcher.on_match {
174                    OnMatch::Action(action) => {
175                        result = Some(action.clone());
176                        Some(OnMatchTrace::Action(action.clone()))
177                    }
178                    OnMatch::Matcher(nested) => {
179                        let nested_trace = nested.evaluate_with_trace(ctx);
180                        result.clone_from(&nested_trace.result);
181                        Some(OnMatchTrace::Nested(Box::new(nested_trace)))
182                    }
183                };
184
185                steps.push(EvalStep {
186                    index,
187                    matched: true,
188                    predicate_trace,
189                    on_match,
190                });
191
192                // First-match-wins: stop if we got a result
193                if result.is_some() {
194                    return EvalTrace {
195                        result,
196                        steps,
197                        used_fallback: false,
198                    };
199                }
200                // Nested matcher returned None → continue to next field_matcher
201            } else {
202                steps.push(EvalStep {
203                    index,
204                    matched: false,
205                    predicate_trace,
206                    on_match: None,
207                });
208            }
209        }
210
211        // Fallback
212        let used_fallback = result.is_none() && self.on_no_match.is_some();
213        if result.is_none() {
214            result = self.on_no_match.as_ref().and_then(|om| match om {
215                OnMatch::Action(a) => Some(a.clone()),
216                OnMatch::Matcher(nested) => nested.evaluate(ctx),
217            });
218        }
219
220        EvalTrace {
221            result,
222            steps,
223            used_fallback,
224        }
225    }
226
227    /// Returns the number of field matchers.
228    pub fn len(&self) -> usize {
229        self.matcher_list.len()
230    }
231
232    /// Returns `true` if there are no field matchers.
233    pub fn is_empty(&self) -> bool {
234        self.matcher_list.is_empty()
235    }
236
237    /// Returns `true` if there is an `on_no_match` fallback.
238    pub fn has_fallback(&self) -> bool {
239        self.on_no_match.is_some()
240    }
241
242    /// Calculate the maximum depth of this matcher tree.
243    ///
244    /// Used for depth limit validation at config time.
245    pub fn depth(&self) -> usize {
246        let field_depth = self
247            .matcher_list
248            .iter()
249            .map(|fm| {
250                let pred_depth = fm.predicate.depth();
251                let nested_depth = match &fm.on_match {
252                    OnMatch::Action(_) => 0,
253                    OnMatch::Matcher(m) => m.depth(),
254                };
255                pred_depth.max(nested_depth)
256            })
257            .max()
258            .unwrap_or(0);
259
260        let no_match_depth = self.on_no_match.as_ref().map_or(0, |om| match om {
261            OnMatch::Action(_) => 0,
262            OnMatch::Matcher(m) => m.depth(),
263        });
264
265        1 + field_depth.max(no_match_depth)
266    }
267
268    /// Validate this matcher against safety constraints.
269    ///
270    /// Checks:
271    /// - Nesting depth does not exceed [`MAX_DEPTH`]
272    ///
273    /// Call this at config load time to catch errors early.
274    ///
275    /// # Errors
276    ///
277    /// Returns [`MatcherError::DepthExceeded`] if nesting is too deep.
278    pub fn validate(&self) -> Result<(), MatcherError> {
279        let depth = self.depth();
280        if depth > MAX_DEPTH {
281            return Err(MatcherError::DepthExceeded {
282                depth,
283                max: MAX_DEPTH,
284            });
285        }
286        Ok(())
287    }
288}

292        f.debug_struct("Matcher")
293            .field("matcher_list_len", &self.matcher_list.len())
294            .field("has_fallback", &self.on_no_match.is_some())
295            .finish()
296    }

733        let cases: Vec<(Matcher<TestCtx, String>, TestCtx)> = vec![

736                Matcher::new(vec![create_field_matcher("hello", "hit")], None),
737                TestCtx {

739                },

741            // No match, no fallback

743                Matcher::new(vec![create_field_matcher("nope", "hit")], None),
744                TestCtx {
745                    value: "hello".into(),
746                },
747            ),
748            // No match, with fallback
749            (
750                Matcher::new(
751                    vec![create_field_matcher("nope", "hit")],
752                    Some(OnMatch::action("fallback".into())),
753                ),
754                TestCtx {
755                    value: "hello".into(),
756                },
757            ),
758            // Empty matcher
759            (
760                Matcher::empty(),
761                TestCtx {
762                    value: "hello".into(),
763                },
764            ),
765        ];
766

771                eval_result, trace.result,

774        }
775    }
