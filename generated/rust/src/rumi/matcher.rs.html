1//! Matcher — Top-level matcher with first-match-wins semantics
2//!
3//! The `Matcher` is the entry point for evaluation. It contains a list of
4//! field matchers and evaluates them in order, returning the first match.
5
6use crate::{EvalStep, EvalTrace, FieldMatcher, MatcherError, OnMatch, OnMatchTrace, MAX_DEPTH};
7use std::fmt::Debug;
8use std::marker::PhantomData;
9
10/// Top-level matcher with first-match-wins semantics.
11///
12/// A Matcher contains:
13/// - A list of field matchers (predicate + action pairs)
14/// - An optional `on_no_match` fallback
15///
16/// Evaluation iterates through the field matchers in order and returns
17/// the action from the first matching predicate. If no predicate matches,
18/// the `on_no_match` action is returned (if present).
19///
20/// # Type Parameters
21///
22/// - `Ctx`: The context type to match against
23/// - `A`: The action type (must be `Clone + Send + Sync + 'static`)
24///
25/// # INV (Dijkstra): First-match-wins
26///
27/// Field matchers are evaluated in order. The first matching predicate
28/// terminates evaluation, even if later predicates would also match.
29///
30/// # xDS Semantics: Nested Matcher Failure Propagates
31///
32/// When an `OnMatch` contains a nested matcher and that nested matcher
33/// returns no match, the ENTIRE `OnMatch` fails — there is no fallback
34/// to a sibling action (because `OnMatch` is exclusive per xDS proto).
35///
36/// # Example
37///
38/// ```ignore
39/// let matcher = Matcher::new(
40///     vec![
41///         FieldMatcher::new(api_path_predicate, OnMatch::action("api".to_string())),
42///         FieldMatcher::new(static_path_predicate, OnMatch::action("static".to_string())),
43///     ],
44///     Some(OnMatch::action("default".to_string())),
45/// );
46///
47/// let action = matcher.evaluate(&request);
48/// ```
49pub struct Matcher<Ctx, A: Clone + Send + Sync + 'static> {
50    /// The list of field matchers to evaluate.
51    pub matcher_list: Vec<FieldMatcher<Ctx, A>>,
52
53    /// Fallback when no field matcher matches.
54    /// Note: per xDS, this is at the Matcher level, not per-OnMatch.
55    pub on_no_match: Option<OnMatch<Ctx, A>>,
56
57    _phantom: PhantomData<Ctx>,
58}
59
60impl<Ctx, A: Clone + Send + Sync + 'static> Matcher<Ctx, A> {
61    /// Create a new matcher.
62    pub fn new(
63        matcher_list: Vec<FieldMatcher<Ctx, A>>,
64        on_no_match: Option<OnMatch<Ctx, A>>,
65    ) -> Self {
66        Self {
67            matcher_list,
68            on_no_match,
69            _phantom: PhantomData,
70        }
71    }
72
73    /// Create an empty matcher (no field matchers, no fallback).
74    #[must_use]
75    pub fn empty() -> Self {
76        Self {
77            matcher_list: Vec::new(),
78            on_no_match: None,
79            _phantom: PhantomData,
80        }
81    }
82
83    /// Evaluate this matcher against the given context.
84    ///
85    /// Returns the action from the first matching field matcher,
86    /// or the `on_no_match` action if nothing matches.
87    ///
88    /// # First-match-wins semantics (INV)
89    ///
90    /// Field matchers are evaluated in order. Evaluation stops at the
91    /// first matching predicate.
92    ///
93    /// # xDS Nested Matcher Semantics
94    ///
95    /// If a field matcher's `OnMatch` contains a nested matcher:
96    /// - Evaluation continues into the nested matcher
97    /// - If the nested matcher returns Some(action), that's returned
98    /// - If the nested matcher returns None, this field matcher is considered
99    ///   NOT matched, and we continue to the next field matcher
100    ///
101    /// This matches xDS semantics where nested matcher failure propagates up.
102    pub fn evaluate(&self, ctx: &Ctx) -> Option<A> {
103        // First-match-wins: iterate through matchers, return first match
104        for field_matcher in &self.matcher_list {
105            if field_matcher.matches(ctx) {
106                // OnMatch is now an enum: either Action or Matcher
107                match &field_matcher.on_match {
108                    OnMatch::Action(action) => return Some(action.clone()),
109                    OnMatch::Matcher(nested) => {
110                        // xDS semantics: nested matcher failure propagates
111                        // If nested returns None, continue to next field_matcher
112                        if let Some(action) = nested.evaluate(ctx) {
113                            return Some(action);
114                        }
115                        // Nested returned None → this field_matcher is NOT a match
116                        // Continue to next field_matcher
117                    }
118                }
119            }
120        }
121
122        // No match: return on_no_match action if present
123        self.on_no_match.as_ref().and_then(|om| match om {
124            OnMatch::Action(a) => Some(a.clone()),
125            OnMatch::Matcher(nested) => nested.evaluate(ctx),
126        })
127    }
128
129    /// Evaluate with full trace for debugging.
130    ///
131    /// Returns the same result as [`evaluate()`](Self::evaluate) plus
132    /// the full evaluation path: which field matchers were checked, which
133    /// predicates fired, and whether the fallback was used.
134    ///
135    /// # INV: `result` == `evaluate()` result
136    ///
137    /// The returned [`EvalTrace::result`] always equals what `evaluate()`
138    /// would return for the same context.
139    #[must_use]
140    pub fn evaluate_with_trace(&self, ctx: &Ctx) -> EvalTrace<A> {
141        let mut steps = Vec::new();
142        let mut result = None;
143
144        for (index, field_matcher) in self.matcher_list.iter().enumerate() {
145            let predicate_trace = field_matcher.predicate.evaluate_with_trace(ctx);
146            let pred_matched = predicate_trace.matched();
147
148            if pred_matched {
149                let on_match = match &field_matcher.on_match {
150                    OnMatch::Action(action) => {
151                        result = Some(action.clone());
152                        Some(OnMatchTrace::Action(action.clone()))
153                    }
154                    OnMatch::Matcher(nested) => {
155                        let nested_trace = nested.evaluate_with_trace(ctx);
156                        result.clone_from(&nested_trace.result);
157                        Some(OnMatchTrace::Nested(Box::new(nested_trace)))
158                    }
159                };
160
161                steps.push(EvalStep {
162                    index,
163                    matched: true,
164                    predicate_trace,
165                    on_match,
166                });
167
168                // First-match-wins: stop if we got a result
169                if result.is_some() {
170                    return EvalTrace {
171                        result,
172                        steps,
173                        used_fallback: false,
174                    };
175                }
176                // Nested matcher returned None → continue to next field_matcher
177            } else {
178                steps.push(EvalStep {
179                    index,
180                    matched: false,
181                    predicate_trace,
182                    on_match: None,
183                });
184            }
185        }
186
187        // Fallback
188        let used_fallback = result.is_none() && self.on_no_match.is_some();
189        if result.is_none() {
190            result = self.on_no_match.as_ref().and_then(|om| match om {
191                OnMatch::Action(a) => Some(a.clone()),
192                OnMatch::Matcher(nested) => nested.evaluate(ctx),
193            });
194        }
195
196        EvalTrace {
197            result,
198            steps,
199            used_fallback,
200        }
201    }
202
203    /// Returns the number of field matchers.
204    pub fn len(&self) -> usize {
205        self.matcher_list.len()
206    }
207
208    /// Returns `true` if there are no field matchers.
209    pub fn is_empty(&self) -> bool {
210        self.matcher_list.is_empty()
211    }
212
213    /// Returns `true` if there is an `on_no_match` fallback.
214    pub fn has_fallback(&self) -> bool {
215        self.on_no_match.is_some()
216    }
217
218    /// Calculate the maximum depth of this matcher tree.
219    ///
220    /// Used for depth limit validation at config time.
221    pub fn depth(&self) -> usize {
222        let field_depth = self
223            .matcher_list
224            .iter()
225            .map(|fm| {
226                let pred_depth = fm.predicate.depth();
227                let nested_depth = match &fm.on_match {
228                    OnMatch::Action(_) => 0,
229                    OnMatch::Matcher(m) => m.depth(),
230                };
231                pred_depth.max(nested_depth)
232            })
233            .max()
234            .unwrap_or(0);
235
236        let no_match_depth = self.on_no_match.as_ref().map_or(0, |om| match om {
237            OnMatch::Action(_) => 0,
238            OnMatch::Matcher(m) => m.depth(),
239        });
240
241        1 + field_depth.max(no_match_depth)
242    }
243
244    /// Validate this matcher against safety constraints.
245    ///
246    /// Checks:
247    /// - Nesting depth does not exceed [`MAX_DEPTH`]
248    ///
249    /// Call this at config load time to catch errors early.
250    ///
251    /// # Errors
252    ///
253    /// Returns [`MatcherError::DepthExceeded`] if nesting is too deep.
254    pub fn validate(&self) -> Result<(), MatcherError> {
255        let depth = self.depth();
256        if depth > MAX_DEPTH {
257            return Err(MatcherError::DepthExceeded {
258                depth,
259                max: MAX_DEPTH,
260            });
261        }
262        Ok(())
263    }
264}

266impl<Ctx, A: Clone + Send + Sync + Debug + 'static> Debug for Matcher<Ctx, A> {
267    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
268        f.debug_struct("Matcher")
269            .field("matcher_list_len", &self.matcher_list.len())
270            .field("has_fallback", &self.on_no_match.is_some())
271            .finish()
272    }
273}
