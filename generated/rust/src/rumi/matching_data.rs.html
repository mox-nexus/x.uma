1//! `MatchingData` — Type-erased data that flows between `DataInput` and `InputMatcher`
2//!
3//! This is the key insight from Envoy's design: type erasure at the data level.
4//! `DataInputs` produce `MatchingData`, and `InputMatchers` consume it.
5//! This allows `InputMatchers` to be non-generic and shareable across contexts.
6//!
7//! # Extensibility via `Custom`
8//!
9//! For domain-specific types not covered by the primitives, implement
10//! [`CustomMatchData`] and wrap in `MatchingData::Custom(Arc::new(your_type))`.
11
12use std::any::Any;
13use std::fmt::Debug;
14use std::sync::Arc;
15
16/// Extension trait for custom matching data types.
17///
18/// This enables users to extend `MatchingData` without modifying the core.
19/// Implement this trait for domain-specific types, then wrap with
20/// `MatchingData::Custom(Arc::new(your_type))`.
21///
22/// # Thread Safety
23///
24/// Implementations must be `Send + Sync` to support concurrent evaluation
25/// and FFI use cases.
26///
27/// # Example
28///
29/// ```
30/// use std::any::Any;
31/// use std::sync::Arc;
32/// use rumi::{CustomMatchData, MatchingData};
33///
34/// #[derive(Debug)]
35/// struct GeoLocation {
36///     lat: f64,
37///     lon: f64,
38/// }
39///
40/// impl CustomMatchData for GeoLocation {
41///     fn custom_type_name(&self) -> &'static str {
42///         "geo_location"
43///     }
44///
45///     fn as_any(&self) -> &dyn Any {
46///         self
47///     }
48/// }
49///
50/// let data = MatchingData::Custom(Arc::new(GeoLocation { lat: 37.7749, lon: -122.4194 }));
51/// assert!(data.is_custom());
52/// assert_eq!(data.type_name(), "geo_location");
53/// ```
54pub trait CustomMatchData: Send + Sync + Debug {
55    /// Returns a human-readable type identifier.
56    ///
57    /// Used for config-time validation to check if a `DataInput`
58    /// produces compatible data for an `InputMatcher`.
59    ///
60    /// Convention: use `snake_case` names, e.g., `"geo_location"`, `"jwt_claims"`.
61    ///
62    /// Note: Named `custom_type_name` to avoid collision with `Any::type_id`.
63    fn custom_type_name(&self) -> &'static str;
64
65    /// Returns a reference to `self` as `&dyn Any`.
66    ///
67    /// Enables downcasting in `InputMatcher` implementations:
68    ///
69    /// ```ignore
70    /// if let Some(geo) = custom.as_any().downcast_ref::<GeoLocation>() {
71    ///     // use geo.lat, geo.lon
72    /// }
73    /// ```
74    fn as_any(&self) -> &dyn Any;
75}
76
77/// The erased data type that flows between `DataInput` and `InputMatcher`.
78///
79/// Inspired by Envoy's `MatchingDataType = variant<monostate, string, shared_ptr<CustomMatchData>>`.
80///
81/// # Variants
82///
83/// - `None` — No data available (extractor returned nothing)
84/// - `String` — String data (most common: headers, paths, query params)
85/// - `Int` — Integer data
86/// - `Bool` — Boolean data
87/// - `Bytes` — Raw bytes data
88/// - `Custom` — User-defined types implementing [`CustomMatchData`]
89///
90/// # Hybrid Design
91///
92/// Primitives stay stack-allocated (fast path), while `Custom` provides
93/// extensibility via heap-allocated trait objects. This mirrors Envoy's
94/// production-proven approach.
95///
96/// # Example
97///
98/// ```
99/// use rumi::MatchingData;
100///
101/// let data = MatchingData::String("hello".to_string());
102/// assert_eq!(data.as_str(), Some("hello"));
103/// assert!(!data.is_none());
104/// ```
105#[derive(Debug, Clone)]
106pub enum MatchingData {
107    /// No data available (extractor returned nothing).
108    /// When a predicate receives this, it evaluates to `false` (INV: Dijkstra).
109    None,
110
111    /// String data — the most common case for HTTP headers, paths, etc.
112    String(String),
113
114    /// Integer data.
115    Int(i64),
116
117    /// Boolean data.
118    Bool(bool),
119
120    /// Raw bytes data.
121    Bytes(Vec<u8>),
122
123    /// Custom data type for domain-specific extensions.
124    ///
125    /// Wrap your [`CustomMatchData`] implementation with `Arc`:
126    /// ```
127    /// use std::sync::Arc;
128    /// use rumi::{CustomMatchData, MatchingData};
129    /// # use std::any::Any;
130    /// # #[derive(Debug)] struct MyType;
131    /// # impl CustomMatchData for MyType {
132    /// #     fn custom_type_name(&self) -> &'static str { "my_type" }
133    /// #     fn as_any(&self) -> &dyn Any { self }
134    /// # }
135    ///
136    /// let data = MatchingData::Custom(Arc::new(MyType));
137    /// ```
138    Custom(Arc<dyn CustomMatchData>),
139}
140
141// Manual PartialEq implementation because trait objects don't auto-derive it.
142// For Custom variants, we use Arc pointer equality (same allocation = equal).
143impl PartialEq for MatchingData {
144    fn eq(&self, other: &Self) -> bool {
145        match (self, other) {
146            (Self::None, Self::None) => true,
147            (Self::String(a), Self::String(b)) => a == b,
148            (Self::Int(a), Self::Int(b)) => a == b,
149            (Self::Bool(a), Self::Bool(b)) => a == b,
150            (Self::Bytes(a), Self::Bytes(b)) => a == b,
151            (Self::Custom(a), Self::Custom(b)) => Arc::ptr_eq(a, b),
152            _ => false,
153        }
154    }
155}

157impl MatchingData {
158    /// Returns `true` if this is the `None` variant.
159    ///
160    /// # Example
161    ///
162    /// ```
163    /// use rumi::MatchingData;
164    ///
165    /// assert!(MatchingData::None.is_none());
166    /// assert!(!MatchingData::String("x".to_string()).is_none());
167    /// ```
168    #[inline]
169    #[must_use]
170    pub fn is_none(&self) -> bool {
171        matches!(self, Self::None)
172    }

209    /// Try to get the value as a string slice.
210    ///
211    /// # Example
212    ///
213    /// ```
214    /// use rumi::MatchingData;
215    ///
216    /// let data = MatchingData::String("hello".to_string());
217    /// assert_eq!(data.as_str(), Some("hello"));
218    ///
219    /// let data = MatchingData::Int(42);
220    /// assert_eq!(data.as_str(), None);
221    /// ```
222    #[inline]
223    #[must_use]
224    pub fn as_str(&self) -> Option<&str> {
225        match self {
226            Self::String(s) => Some(s.as_str()),
227            _ => Option::None,
228        }
229    }

231    /// Try to get the value as an integer.
232    #[inline]
233    #[must_use]
234    pub fn as_int(&self) -> Option<i64> {
235        match self {
236            Self::Int(i) => Some(*i),
237            _ => Option::None,
238        }
239    }

246            Self::Bool(b) => Some(*b),
247            _ => Option::None,

254    pub fn as_bytes(&self) -> Option<&[u8]> {
255        match self {

261    /// Try to get the value as a custom match data reference.
262    ///
263    /// Returns a reference to the inner [`CustomMatchData`] trait object.
264    /// Use [`CustomMatchData::as_any`] to downcast to the concrete type.
265    ///
266    /// # Example
267    ///
268    /// ```
269    /// use std::sync::Arc;
270    /// use std::any::Any;
271    /// use rumi::{CustomMatchData, MatchingData};
272    ///
273    /// #[derive(Debug)]
274    /// struct MyData(i32);
275    ///
276    /// impl CustomMatchData for MyData {
277    ///     fn custom_type_name(&self) -> &'static str { "my_data" }
278    ///     fn as_any(&self) -> &dyn Any { self }
279    /// }
280    ///
281    /// let data = MatchingData::Custom(Arc::new(MyData(42)));
282    /// if let Some(custom) = data.as_custom() {
283    ///     let my_data = custom.as_any().downcast_ref::<MyData>().unwrap();
284    ///     assert_eq!(my_data.0, 42);
285    /// }
286    /// ```
287    #[inline]
288    #[must_use]
289    pub fn as_custom(&self) -> Option<&dyn CustomMatchData> {
290        match self {
291            Self::Custom(c) => Some(c.as_ref()),
292            _ => Option::None,
293        }
294    }
295
296    /// Returns a string describing the type of this data.
297    ///
298    /// Useful for config-time validation when checking if a `DataInput`
299    /// produces compatible data for an `InputMatcher`.
300    ///
301    /// For `Custom` variants, this delegates to [`CustomMatchData::custom_type_name`].
302    #[must_use]
303    pub fn type_name(&self) -> &'static str {
304        match self {
305            Self::None => "none",
306            Self::String(_) => "string",
307            Self::Int(_) => "int",
308            Self::Bool(_) => "bool",
309            Self::Bytes(_) => "bytes",
310            Self::Custom(c) => c.custom_type_name(),
311        }
312    }

512    /// Integration test: Full extension story with custom types
513    ///
514    /// This test proves that users can extend the system without touching core:
515    /// 1. Custom domain type (GeoLocation)
516    /// 2. Custom context (GeoRequest)
517    /// 3. Custom DataInput (LocationInput)
518    /// 4. Custom InputMatcher (WithinRadiusMatcher)
519    /// 5. Full Matcher pipeline with actions
520    mod integration {
521        use super::*;
522        use crate::{
523            DataInput, FieldMatcher, InputMatcher, Matcher, OnMatch, Predicate, SinglePredicate,
524        };
525
526        // ══════════════════════════════════════════════════════════════════════
527        // Step 1: Define a custom domain type (extension point)
528        // ══════════════════════════════════════════════════════════════════════
529
530        #[derive(Debug, Clone)]
531        struct GeoLocation {
532            lat: f64,
533            lon: f64,
534        }
535
536        impl GeoLocation {
537            fn distance_to(&self, other: &GeoLocation) -> f64 {
538                // Simplified distance (not real haversine, just for testing)
539                ((self.lat - other.lat).powi(2) + (self.lon - other.lon).powi(2)).sqrt()
540            }
541        }
542
543        impl CustomMatchData for GeoLocation {
544            fn custom_type_name(&self) -> &'static str {
545                "geo_location"
546            }
547
548            fn as_any(&self) -> &dyn Any {
549                self
550            }
551        }
552
553        // ══════════════════════════════════════════════════════════════════════
554        // Step 2: Define a custom context (the "Ctx" generic parameter)
555        // ══════════════════════════════════════════════════════════════════════
556
557        #[derive(Debug)]
558        struct GeoRequest {
559            user_location: GeoLocation,
560            user_id: String,
561        }
562
563        // ══════════════════════════════════════════════════════════════════════
564        // Step 3: Define a custom DataInput (extracts GeoLocation from context)
565        // ══════════════════════════════════════════════════════════════════════
566
567        #[derive(Debug)]
568        struct LocationInput;
569
570        impl DataInput<GeoRequest> for LocationInput {
571            fn get(&self, ctx: &GeoRequest) -> MatchingData {
572                MatchingData::Custom(Arc::new(ctx.user_location.clone()))
573            }
574
575            fn data_type(&self) -> &'static str {
576                "geo_location"
577            }
578        }
579
580        // ══════════════════════════════════════════════════════════════════════
581        // Step 4: Define a custom InputMatcher (matches on geo proximity)
582        // ══════════════════════════════════════════════════════════════════════
583
584        #[derive(Debug)]
585        struct WithinRadiusMatcher {
586            center: GeoLocation,
587            radius: f64,
588        }
589
590        impl WithinRadiusMatcher {
591            fn new(center: GeoLocation, radius: f64) -> Self {
592                Self { center, radius }
593            }
594        }
595
596        impl InputMatcher for WithinRadiusMatcher {
597            fn matches(&self, value: &MatchingData) -> bool {
598                // Get custom data, downcast to GeoLocation, check distance
599                value
600                    .as_custom()
601                    .and_then(|c| c.as_any().downcast_ref::<GeoLocation>())
602                    .is_some_and(|loc| loc.distance_to(&self.center) <= self.radius)
603            }
604
605            fn supported_types(&self) -> &[&'static str] {
606                &["geo_location"]
607            }
608        }
609
610        // ══════════════════════════════════════════════════════════════════════
611        // Step 5: Integration test - wire everything together
612        // ══════════════════════════════════════════════════════════════════════
613
614        #[test]
615        fn test_full_custom_extension_pipeline() {
616            // Define regions as geo-fences
617            let sf_center = GeoLocation {
618                lat: 37.7749,
619                lon: -122.4194,
620            };
621            let nyc_center = GeoLocation {
622                lat: 40.7128,
623                lon: -74.0060,
624            };
625
626            // Build matcher: route to different backends based on location
627            let matcher: Matcher<GeoRequest, String> = Matcher::new(
628                vec![
629                    // Rule 1: SF area → sf_backend
630                    FieldMatcher::new(
631                        Predicate::Single(SinglePredicate::new(
632                            Box::new(LocationInput),
633                            Box::new(WithinRadiusMatcher::new(sf_center.clone(), 1.0)),
634                        )),
635                        OnMatch::Action("sf_backend".to_string()),
636                    ),
637                    // Rule 2: NYC area → nyc_backend
638                    FieldMatcher::new(
639                        Predicate::Single(SinglePredicate::new(
640                            Box::new(LocationInput),
641                            Box::new(WithinRadiusMatcher::new(nyc_center.clone(), 1.0)),
642                        )),
643                        OnMatch::Action("nyc_backend".to_string()),
644                    ),
645                ],
646                Some(OnMatch::Action("default_backend".to_string())),
647            );
648
649            // Test 1: User in SF → sf_backend
650            let sf_request = GeoRequest {
651                user_location: GeoLocation {
652                    lat: 37.78,
653                    lon: -122.42,
654                },
655                user_id: "user_sf".to_string(),
656            };
657            assert_eq!(
658                matcher.evaluate(&sf_request),
659                Some("sf_backend".to_string())
660            );
661
662            // Test 2: User in NYC → nyc_backend
663            let nyc_request = GeoRequest {
664                user_location: GeoLocation {
665                    lat: 40.71,
666                    lon: -74.01,
667                },
668                user_id: "user_nyc".to_string(),
669            };
670            assert_eq!(
671                matcher.evaluate(&nyc_request),
672                Some("nyc_backend".to_string())
673            );
674
675            // Test 3: User in London → default_backend (no match)
676            let london_request = GeoRequest {
677                user_location: GeoLocation {
678                    lat: 51.5074,
679                    lon: -0.1278,
680                },
681                user_id: "user_london".to_string(),
682            };
683            assert_eq!(
684                matcher.evaluate(&london_request),
685                Some("default_backend".to_string())
686            );
687        }
688    }
