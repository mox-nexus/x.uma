1//! Hook event types and context for Claude Code.
2
3use std::collections::HashMap;
4
5/// All Claude Code hook event types.
6#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
7pub enum HookEvent {
8    /// Before a tool executes.
9    PreToolUse,
10    /// After a tool completes.
11    PostToolUse,
12    /// Main agent considers stopping.
13    Stop,
14    /// Subagent considers stopping.
15    SubagentStop,
16    /// User submits a prompt.
17    UserPromptSubmit,
18    /// Session begins.
19    SessionStart,
20    /// Session ends.
21    SessionEnd,
22    /// Before context compaction.
23    PreCompact,
24    /// Notification sent to user.
25    Notification,
26}
27
28impl HookEvent {
29    /// Convert to string representation.
30    #[must_use]
31    pub fn as_str(&self) -> &'static str {
32        match self {
33            Self::PreToolUse => "PreToolUse",
34            Self::PostToolUse => "PostToolUse",
35            Self::Stop => "Stop",
36            Self::SubagentStop => "SubagentStop",
37            Self::UserPromptSubmit => "UserPromptSubmit",
38            Self::SessionStart => "SessionStart",
39            Self::SessionEnd => "SessionEnd",
40            Self::PreCompact => "PreCompact",
41            Self::Notification => "Notification",
42        }
43    }
44}
45
46/// Claude Code hook context for matching.
47///
48/// Carries all data available at the hook enforcement point:
49/// event type, tool information, and session metadata.
50#[derive(Debug, Clone)]
51pub struct HookContext {
52    event: HookEvent,
53    tool_name: String,
54    arguments: HashMap<String, String>,
55    session_id: String,
56    cwd: String,
57    git_branch: Option<String>,
58}
59
60impl HookContext {
61    /// Create a `PreToolUse` hook context.
62    #[must_use]
63    pub fn pre_tool_use(tool_name: impl Into<String>) -> Self {
64        Self {
65            event: HookEvent::PreToolUse,
66            tool_name: tool_name.into(),
67            arguments: HashMap::new(),
68            session_id: String::new(),
69            cwd: String::new(),
70            git_branch: None,
71        }
72    }
73
74    /// Create a `PostToolUse` hook context.
75    #[must_use]
76    pub fn post_tool_use(tool_name: impl Into<String>) -> Self {
77        Self {
78            event: HookEvent::PostToolUse,
79            tool_name: tool_name.into(),
80            arguments: HashMap::new(),
81            session_id: String::new(),
82            cwd: String::new(),
83            git_branch: None,
84        }
85    }
86
87    /// Create a `Stop` hook context.
88    #[must_use]
89    pub fn stop() -> Self {
90        Self::eventonly(HookEvent::Stop)
91    }
92
93    /// Create a `SubagentStop` hook context.
94    #[must_use]
95    pub fn subagent_stop() -> Self {
96        Self::eventonly(HookEvent::SubagentStop)
97    }
98
99    /// Create a `UserPromptSubmit` hook context.
100    #[must_use]
101    pub fn user_prompt_submit() -> Self {
102        Self::eventonly(HookEvent::UserPromptSubmit)
103    }
104
105    /// Create a `SessionStart` hook context.
106    #[must_use]
107    pub fn session_start() -> Self {
108        Self::eventonly(HookEvent::SessionStart)
109    }
110
111    /// Create a `SessionEnd` hook context.
112    #[must_use]
113    pub fn session_end() -> Self {
114        Self::eventonly(HookEvent::SessionEnd)
115    }
116
117    /// Create a `PreCompact` hook context.
118    #[must_use]
119    pub fn pre_compact() -> Self {
120        Self::eventonly(HookEvent::PreCompact)
121    }
122
123    /// Create a `Notification` hook context.
124    #[must_use]
125    pub fn notification() -> Self {
126        Self::eventonly(HookEvent::Notification)
127    }
128
129    /// Internal: create a context with only an event (no tool).
130    fn eventonly(event: HookEvent) -> Self {
131        Self {
132            event,
133            tool_name: String::new(),
134            arguments: HashMap::new(),
135            session_id: String::new(),
136            cwd: String::new(),
137            git_branch: None,
138        }
139    }
140
141    /// Add a tool argument (builder pattern).
142    #[must_use]
143    pub fn with_arg(mut self, name: impl Into<String>, value: impl Into<String>) -> Self {
144        self.arguments.insert(name.into(), value.into());
145        self
146    }
147
148    /// Set the session ID (builder pattern).
149    #[must_use]
150    pub fn with_session_id(mut self, id: impl Into<String>) -> Self {
151        self.session_id = id.into();
152        self
153    }
154
155    /// Set the current working directory (builder pattern).
156    #[must_use]
157    pub fn with_cwd(mut self, cwd: impl Into<String>) -> Self {
158        self.cwd = cwd.into();
159        self
160    }
161
162    /// Set the git branch (builder pattern).
163    #[must_use]
164    pub fn with_git_branch(mut self, branch: impl Into<String>) -> Self {
165        self.git_branch = Some(branch.into());
166        self
167    }
168
169    /// Get the hook event type.
170    #[must_use]
171    pub fn event(&self) -> HookEvent {
172        self.event
173    }
174
175    /// Get the tool name (empty string for non-tool events).
176    #[must_use]
177    pub fn tool_name(&self) -> &str {
178        &self.tool_name
179    }
180
181    /// Get a tool argument by name.
182    #[must_use]
183    pub fn argument(&self, name: &str) -> Option<&str> {
184        self.arguments.get(name).map(String::as_str)
185    }
186
187    /// Get the session ID.
188    #[must_use]
189    pub fn session_id(&self) -> &str {
190        &self.session_id
191    }
192
193    /// Get the current working directory.
194    #[must_use]
195    pub fn cwd(&self) -> &str {
196        &self.cwd
197    }
198
199    /// Get the git branch (if in a repository).
200    #[must_use]
201    pub fn git_branch(&self) -> Option<&str> {
202        self.git_branch.as_deref()
203    }
204}
