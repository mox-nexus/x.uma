1//! User-facing configuration types for Claude Code hook matching.
2//!
3//! These types provide a declarative API for specifying match rules.
4//! The [compiler](crate::compiler) translates them into runtime `Matcher` trees.
5
6use crate::context::HookEvent;
7use rumi::prelude::*;
8use std::fmt;
9
10/// How to match a string value.
11#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
12pub enum StringMatch {
13    /// Exact equality.
14    Exact(String),
15    /// Starts with prefix.
16    Prefix(String),
17    /// Ends with suffix.
18    Suffix(String),
19    /// Contains substring.
20    Contains(String),
21    /// Matches regular expression (Rust `regex` crate syntax).
22    Regex(String),
23}
24
25impl StringMatch {
26    /// Compile this match config into a runtime `InputMatcher`.
27    ///
28    /// # Errors
29    ///
30    /// Returns `CompileError` if the pattern is an invalid regular expression.
31    pub fn to_input_matcher(&self) -> Result<Box<dyn InputMatcher>, CompileError> {
32        match self {
33            Self::Exact(v) => Ok(Box::new(ExactMatcher::new(v.as_str()))),
34            Self::Prefix(v) => Ok(Box::new(PrefixMatcher::new(v.as_str()))),
35            Self::Suffix(v) => Ok(Box::new(SuffixMatcher::new(v.as_str()))),
36            Self::Contains(v) => Ok(Box::new(ContainsMatcher::new(v.as_str()))),
37            Self::Regex(v) => StringMatcher::regex(v)
38                .map(|sm| Box::new(sm) as Box<dyn InputMatcher>)
39                .map_err(|e| CompileError(format!("invalid regex `{v}`: {e}"))),
40        }
41    }
42}
43
44impl fmt::Display for StringMatch {
45    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
46        match self {
47            Self::Exact(v) => write!(f, "Exact(\"{v}\")"),
48            Self::Prefix(v) => write!(f, "Prefix(\"{v}\")"),
49            Self::Suffix(v) => write!(f, "Suffix(\"{v}\")"),
50            Self::Contains(v) => write!(f, "Contains(\"{v}\")"),
51            Self::Regex(v) => write!(f, "Regex(\"{v}\")"),
52        }
53    }
54}

56/// User-friendly configuration for matching Claude Code hook events.
57///
58/// All fields are optional. Omitted fields match anything.
59/// All present fields are `ANDed` (every condition must match).
60///
61/// # Example
62///
63/// ```ignore
64/// use rumi_claude::prelude::*;
65///
66/// let rule = HookMatch {
67///     event: Some(HookEvent::PreToolUse),
68///     tool_name: Some(StringMatch::Exact("Bash".into())),
69///     arguments: Some(vec![
70///         ArgumentMatch {
71///             name: "command".into(),
72///             value: StringMatch::Contains("rm -rf".into()),
73///         },
74///     ]),
75///     ..Default::default()
76/// };
77/// ```
78#[derive(Debug, Clone, Default, serde::Serialize, serde::Deserialize)]
79#[serde(deny_unknown_fields)]
80pub struct HookMatch {
81    /// Match on hook event type (exact match).
82    pub event: Option<HookEvent>,
83    /// Match on tool name.
84    pub tool_name: Option<StringMatch>,
85    /// Match on specific tool arguments (all `ANDed`).
86    pub arguments: Option<Vec<ArgumentMatch>>,
87    /// Match on current working directory.
88    pub cwd: Option<StringMatch>,
89    /// Match on git branch.
90    pub git_branch: Option<StringMatch>,
91}
92
93/// Match a specific tool argument by name.
94#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
95#[serde(deny_unknown_fields)]
96pub struct ArgumentMatch {
97    /// The argument name to extract.
98    pub name: String,
99    /// How to match the argument's value.
100    pub value: StringMatch,
101}
102
103/// Error when compiling a `HookMatch` into a `Matcher`.
104#[derive(Debug)]
105pub struct CompileError(pub(crate) String);
106
107impl fmt::Display for CompileError {
108    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
109        write!(f, "hook match compile error: {}", self.0)
110    }
111}
112
113impl std::error::Error for CompileError {}

115#[cfg(test)]
116mod tests {
117    use super::*;
118
119    #[test]
120    fn string_match_exact_compiles() {
121        let m = StringMatch::Exact("hello".into());
122        let matcher = m.to_input_matcher().unwrap();
123        assert!(matcher.matches(&MatchingData::String("hello".into())));
124        assert!(!matcher.matches(&MatchingData::String("world".into())));
125    }
126
127    #[test]
128    fn string_match_prefix_compiles() {
129        let m = StringMatch::Prefix("/api".into());
130        let matcher = m.to_input_matcher().unwrap();
131        assert!(matcher.matches(&MatchingData::String("/api/users".into())));
132        assert!(!matcher.matches(&MatchingData::String("/other".into())));
133    }
134
135    #[test]
136    fn string_match_suffix_compiles() {
137        let m = StringMatch::Suffix(".rs".into());
138        let matcher = m.to_input_matcher().unwrap();
139        assert!(matcher.matches(&MatchingData::String("main.rs".into())));
140        assert!(!matcher.matches(&MatchingData::String("main.py".into())));
141    }
142
143    #[test]
144    fn string_match_contains_compiles() {
145        let m = StringMatch::Contains("rm -rf".into());
146        let matcher = m.to_input_matcher().unwrap();
147        assert!(matcher.matches(&MatchingData::String("sudo rm -rf /".into())));
148        assert!(!matcher.matches(&MatchingData::String("ls -la".into())));
149    }
150
151    #[test]
152    fn string_match_regex_compiles() {
153        let m = StringMatch::Regex(r"^mcp__.*__delete".into());
154        let matcher = m.to_input_matcher().unwrap();
155        assert!(matcher.matches(&MatchingData::String("mcp__db__delete_row".into())));
156        assert!(!matcher.matches(&MatchingData::String("Write".into())));
157    }
158
159    #[test]
160    fn string_match_invalid_regex_returns_error() {
161        let m = StringMatch::Regex("[invalid".into());
162        let err = m.to_input_matcher().unwrap_err();
163        assert!(err.to_string().contains("invalid regex"));
164    }
165
166    #[test]
167    fn hook_match_default_is_empty() {
168        let m = HookMatch::default();
169        assert!(m.event.is_none());
170        assert!(m.tool_name.is_none());
171        assert!(m.arguments.is_none());
172        assert!(m.cwd.is_none());
173        assert!(m.git_branch.is_none());
174    }
175
176    #[test]
177    fn string_match_display() {
178        assert_eq!(
179            StringMatch::Exact("Bash".into()).to_string(),
180            r#"Exact("Bash")"#
181        );
182        assert_eq!(
183            StringMatch::Regex("^mcp".into()).to_string(),
184            r#"Regex("^mcp")"#
185        );
186    }
