1//! Compiler: Gateway API `HttpRouteMatch` -> rumi Matcher
2//!
3//! Translates user-friendly Gateway API configuration into efficient
4//! runtime matchers operating on `HttpMessage`.
5
6use crate::inputs::{HeaderInput, MethodInput, PathInput, QueryParamInput};
7use crate::message::HttpMessage;
8use k8s_gateway_api::{HttpHeaderMatch, HttpPathMatch, HttpQueryParamMatch, HttpRouteMatch};
9use rumi::prelude::*;
10
11/// Extension trait for compiling `HttpRouteMatch` to rumi Matcher.
12pub trait HttpRouteMatchExt {
13    /// Compile this `HttpRouteMatch` into a rumi Matcher.
14    ///
15    /// The resulting matcher operates on `HttpMessage` and returns
16    /// the provided action when all conditions match.
17    fn compile<A: Clone + Send + Sync + 'static>(&self, action: A) -> Matcher<HttpMessage, A>;
18
19    /// Compile this `HttpRouteMatch` into a Predicate (without action).
20    fn to_predicate(&self) -> Predicate<HttpMessage>;
21}
22
23impl HttpRouteMatchExt for HttpRouteMatch {
24    fn compile<A: Clone + Send + Sync + 'static>(&self, action: A) -> Matcher<HttpMessage, A> {
25        let predicate = self.to_predicate();
26
27        Matcher::new(
28            vec![FieldMatcher::new(predicate, OnMatch::Action(action))],
29            None,
30        )
31    }
32
33    fn to_predicate(&self) -> Predicate<HttpMessage> {
34        let mut predicates: Vec<Predicate<HttpMessage>> = Vec::new();
35
36        // Path matching
37        if let Some(path_match) = &self.path {
38            predicates.push(compile_path_match(path_match));
39        }
40
41        // Method matching
42        if let Some(method) = &self.method {
43            predicates.push(Predicate::Single(SinglePredicate::new(
44                Box::new(MethodInput),
45                Box::new(ExactMatcher::new(method.as_str())),
46            )));
47        }
48
49        // Header matching (all headers are ANDed)
50        if let Some(headers) = &self.headers {
51            for header_match in headers {
52                predicates.push(compile_header_match(header_match));
53            }
54        }
55
56        // Query param matching (all params are ANDed)
57        if let Some(query_params) = &self.query_params {
58            for query_match in query_params {
59                predicates.push(compile_query_param_match(query_match));
60            }
61        }
62
63        // If no conditions, match everything
64        if predicates.is_empty() {
65            return Predicate::Single(SinglePredicate::new(
66                Box::new(PathInput),
67                Box::new(PrefixMatcher::new("")), // Match any path
68            ));
69        }
70
71        // Single predicate or AND them together
72        if predicates.len() == 1 {
73            predicates.pop().unwrap()
74        } else {
75            Predicate::And(predicates)
76        }
77    }
78}

80/// Compile a path match to a predicate.
81fn compile_path_match(path_match: &HttpPathMatch) -> Predicate<HttpMessage> {
82    let input = Box::new(PathInput);
83
84    let matcher: Box<dyn InputMatcher> = match path_match {
85        HttpPathMatch::Exact { value } => Box::new(ExactMatcher::new(value.as_str())),
86        HttpPathMatch::PathPrefix { value } => Box::new(PrefixMatcher::new(value.as_str())),
87        HttpPathMatch::RegularExpression { value } => {
88            Box::new(StringMatcher::regex(value).unwrap_or_else(|_| {
89                // Invalid regex falls back to exact match (fail-safe)
90                StringMatcher::exact(value, false)
91            }))
92        }
93    };
94
95    Predicate::Single(SinglePredicate::new(input, matcher))
96}
97
98/// Compile a header match to a predicate.
99fn compile_header_match(header_match: &HttpHeaderMatch) -> Predicate<HttpMessage> {
100    match header_match {
101        HttpHeaderMatch::Exact { name, value } => {
102            let input = Box::new(HeaderInput::new(name.as_str()));
103            let matcher = Box::new(ExactMatcher::new(value.as_str()));
104            Predicate::Single(SinglePredicate::new(input, matcher))
105        }
106        HttpHeaderMatch::RegularExpression { name, value } => {
107            let input = Box::new(HeaderInput::new(name.as_str()));
108            let matcher: Box<dyn InputMatcher> = Box::new(
109                StringMatcher::regex(value).unwrap_or_else(|_| StringMatcher::exact(value, false)),
110            );
111            Predicate::Single(SinglePredicate::new(input, matcher))
112        }
113    }
114}

116/// Compile a query param match to a predicate.
117fn compile_query_param_match(query_match: &HttpQueryParamMatch) -> Predicate<HttpMessage> {
118    match query_match {
119        HttpQueryParamMatch::Exact { name, value } => {
120            let input = Box::new(QueryParamInput::new(name.as_str()));
121            let matcher = Box::new(ExactMatcher::new(value.as_str()));
122            Predicate::Single(SinglePredicate::new(input, matcher))
123        }
124        HttpQueryParamMatch::RegularExpression { name, value } => {
125            let input = Box::new(QueryParamInput::new(name.as_str()));
126            let matcher: Box<dyn InputMatcher> = Box::new(
127                StringMatcher::regex(value).unwrap_or_else(|_| StringMatcher::exact(value, false)),
128            );
129            Predicate::Single(SinglePredicate::new(input, matcher))
130        }
131    }
132}

134/// Compile multiple `HttpRouteMatch` entries into a single Matcher.
135///
136/// Multiple matches are `ORed` together per Gateway API semantics.
137pub fn compile_route_matches<A: Clone + Send + Sync + 'static>(
138    matches: &[HttpRouteMatch],
139    action: A,
140    on_no_match: Option<A>,
141) -> Matcher<HttpMessage, A> {
142    if matches.is_empty() {
143        // Empty matches = match everything
144        return Matcher::new(
145            vec![FieldMatcher::new(
146                Predicate::Single(SinglePredicate::new(
147                    Box::new(PathInput),
148                    Box::new(PrefixMatcher::new("")),
149                )),
150                OnMatch::Action(action),
151            )],
152            on_no_match.map(OnMatch::Action),
153        );
154    }
155
156    if matches.len() == 1 {
157        let predicate = matches[0].to_predicate();
158        return Matcher::new(
159            vec![FieldMatcher::new(predicate, OnMatch::Action(action))],
160            on_no_match.map(OnMatch::Action),
161        );
162    }
163
164    // Multiple matches: OR them together
165    let predicates: Vec<Predicate<HttpMessage>> = matches
166        .iter()
167        .map(HttpRouteMatchExt::to_predicate)
168        .collect();
169
170    Matcher::new(
171        vec![FieldMatcher::new(
172            Predicate::Or(predicates),
173            OnMatch::Action(action),
174        )],
175        on_no_match.map(OnMatch::Action),
176    )
177}

189    /// Builder for constructing test requests as `HttpMessage`.
190    struct RequestBuilder {
191        headers: Vec<HeaderValue>,
192    }
193
194    impl RequestBuilder {
195        fn new() -> Self {
196            Self { headers: vec![] }
197        }
198
199        fn path(mut self, path: &str) -> Self {
200            self.headers.push(HeaderValue {
201                key: ":path".into(),
202                value: path.into(),
203                raw_value: vec![],
204            });
205            self
206        }
207
208        fn method(mut self, method: &str) -> Self {
209            self.headers.push(HeaderValue {
210                key: ":method".into(),
211                value: method.into(),
212                raw_value: vec![],
213            });
214            self
215        }
216
217        fn header(mut self, name: &str, value: &str) -> Self {
218            self.headers.push(HeaderValue {
219                key: name.to_lowercase(),
220                value: value.into(),
221                raw_value: vec![],
222            });
223            self
224        }
225
226        fn build(self) -> HttpMessage {
227            let req = ProcessingRequest {
228                request: Some(Request::RequestHeaders(HttpHeaders {
229                    headers: Some(HeaderMap {
230                        headers: self.headers,
231                    }),
232                    ..Default::default()
233                })),
234                ..Default::default()
235            };
236            HttpMessage::from(&req)
237        }
238    }
