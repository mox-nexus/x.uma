1//! Typed Python config classes for hook matching.
2//!
3//! These provide IDE autocomplete and type safety on the Python side,
4//! rather than exposing raw dicts (Ace recommendation from guild review).
5
6use pyo3::prelude::*;
7
8/// How to match a string value.
9///
10/// Bare strings passed to `HookMatch` fields are treated as exact matches.
11#[pyclass]
12#[derive(Debug, Clone)]
13pub enum PyStringMatch {
14    /// Exact equality.
15    Exact { value: String },
16    /// Starts with prefix.
17    Prefix { value: String },
18    /// Ends with suffix.
19    Suffix { value: String },
20    /// Contains substring.
21    Contains { value: String },
22    /// Matches regular expression (Rust `regex` crate syntax — linear time).
23    Regex { pattern: String },
24}
25
26#[pymethods]
27impl PyStringMatch {
28    /// Create an exact match.
29    #[staticmethod]
30    fn exact(value: String) -> Self {
31        Self::Exact { value }
32    }
33
34    /// Create a prefix match.
35    #[staticmethod]
36    fn prefix(value: String) -> Self {
37        Self::Prefix { value }
38    }
39
40    /// Create a suffix match.
41    #[staticmethod]
42    fn suffix(value: String) -> Self {
43        Self::Suffix { value }
44    }
45
46    /// Create a contains match.
47    #[staticmethod]
48    fn contains(value: String) -> Self {
49        Self::Contains { value }
50    }
51
52    /// Create a regex match (Rust `regex` crate — guaranteed linear time).
53    #[staticmethod]
54    fn regex(pattern: String) -> Self {
55        Self::Regex { pattern }
56    }
57
58    fn __repr__(&self) -> String {
59        match self {
60            Self::Exact { value } => format!("StringMatch.exact({value:?})"),
61            Self::Prefix { value } => format!("StringMatch.prefix({value:?})"),
62            Self::Suffix { value } => format!("StringMatch.suffix({value:?})"),
63            Self::Contains { value } => format!("StringMatch.contains({value:?})"),
64            Self::Regex { pattern } => format!("StringMatch.regex({pattern:?})"),
65        }
66    }
67}

69/// User-friendly configuration for matching Claude Code hook events.
70///
71/// All fields are optional. Omitted fields match anything.
72/// All present fields are `ANDed` (every condition must match).
73///
74/// # Fail-closed (Vector security requirement)
75///
76/// An empty `HookMatch` (all fields None) is rejected at compile time
77/// unless `match_all=True` is explicitly passed. This prevents accidental
78/// catch-all rules from deserialization bugs.
79#[pyclass]
80#[derive(Debug, Clone)]
81pub struct PyHookMatch {
82    pub(crate) event: Option<String>,
83    pub(crate) tool_name: Option<PyStringMatch>,
84    pub(crate) arguments: Vec<(String, PyStringMatch)>,
85    pub(crate) session_id: Option<PyStringMatch>,
86    pub(crate) cwd: Option<PyStringMatch>,
87    pub(crate) git_branch: Option<PyStringMatch>,
88    pub(crate) match_all: bool,
89}
90
91#[pymethods]
92impl PyHookMatch {
93    /// Create a hook match rule.
94    ///
95    /// Bare strings for `tool_name`, `cwd`, `git_branch` are treated as exact matches.
96    #[new]
97    #[pyo3(signature = (
98        event = None,
99        tool_name = None,
100        arguments = None,
101        session_id = None,
102        cwd = None,
103        git_branch = None,
104        match_all = false,
105    ))]
106    #[allow(clippy::too_many_arguments)]
107    fn new(
108        event: Option<String>,
109        tool_name: Option<PyStringMatchOrStr>,
110        arguments: Option<Vec<(String, PyStringMatchOrStr)>>,
111        session_id: Option<PyStringMatchOrStr>,
112        cwd: Option<PyStringMatchOrStr>,
113        git_branch: Option<PyStringMatchOrStr>,
114        match_all: bool,
115    ) -> Self {
116        Self {
117            event,
118            tool_name: tool_name.map(Into::into),
119            arguments: arguments
120                .unwrap_or_default()
121                .into_iter()
122                .map(|(k, v)| (k, v.into()))
123                .collect(),
124            session_id: session_id.map(Into::into),
125            cwd: cwd.map(Into::into),
126            git_branch: git_branch.map(Into::into),
127            match_all,
128        }
129    }
130
131    fn __repr__(&self) -> String {
132        format!(
133            "HookMatch(event={:?}, tool_name={:?}, match_all={})",
134            self.event, self.tool_name, self.match_all
135        )
136    }
137}

139/// Accept either `PyStringMatch` or a bare `str` (→ exact match).
140///
141/// This enables the Ace-recommended convenience: `tool_name="Bash"`.
142#[derive(Debug, Clone)]
143pub enum PyStringMatchOrStr {
144    Match(PyStringMatch),
145    Str(String),
146}
147
148impl From<PyStringMatchOrStr> for PyStringMatch {
149    fn from(v: PyStringMatchOrStr) -> Self {
150        match v {
151            PyStringMatchOrStr::Match(m) => m,
152            PyStringMatchOrStr::Str(s) => PyStringMatch::Exact { value: s },
153        }
154    }
155}
156
157impl<'py> FromPyObject<'py> for PyStringMatchOrStr {
158    fn extract_bound(ob: &Bound<'py, PyAny>) -> PyResult<Self> {
159        if let Ok(s) = ob.extract::<String>() {
160            Ok(Self::Str(s))
161        } else {
162            Ok(Self::Match(ob.extract::<PyStringMatch>()?))
163        }
164    }
165}
