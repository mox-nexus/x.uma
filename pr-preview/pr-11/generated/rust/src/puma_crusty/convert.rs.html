1//! Conversion from Python config types to Rust types with input validation.
2//!
3//! All validation happens at the FFI boundary (Vector security requirement).
4//! Invalid inputs produce clear Python exceptions, never panics.
5
6use crate::config::{PyHookMatch, PyStringMatch};
7use pyo3::exceptions::PyValueError;
8use pyo3::PyResult;
9use rumi_claude::{ArgumentMatch, HookEvent, HookMatch, StringMatch};
10
11/// Maximum length for any string pattern (8 KB).
12const MAX_PATTERN_LENGTH: usize = 8192;
13
14/// Maximum number of argument matchers per `HookMatch`.
15const MAX_ARGUMENTS: usize = 64;
16
17/// Maximum number of rules per `compile()` call.
18pub const MAX_RULES: usize = 256;
19
20/// Maximum regex pattern length (4 KB).
21const MAX_REGEX_PATTERN_LENGTH: usize = 4096;
22
23/// Convert a Python `PyHookMatch` to a Rust `HookMatch`.
24///
25/// # Errors
26///
27/// Returns `PyValueError` if:
28/// - Empty match without `match_all=True` (V-BYPASS-1: fail-closed)
29/// - Pattern exceeds length limits
30/// - Argument count exceeds limit
31/// - Event string is unrecognized
32pub fn convert_hook_match(py_match: &PyHookMatch) -> PyResult<HookMatch> {
33    // V-BYPASS-1: reject empty matches unless explicitly intended
34    let is_empty = py_match.event.is_none()
35        && py_match.tool_name.is_none()
36        && py_match.arguments.is_empty()
37        && py_match.session_id.is_none()
38        && py_match.cwd.is_none()
39        && py_match.git_branch.is_none();
40
41    if is_empty && !py_match.match_all {
42        return Err(PyValueError::new_err(
43            "empty HookMatch matches everything â€” pass match_all=True to confirm, \
44             or add at least one field to constrain the match",
45        ));
46    }
47
48    // Validate argument count
49    if py_match.arguments.len() > MAX_ARGUMENTS {
50        return Err(PyValueError::new_err(format!(
51            "too many arguments: {} exceeds limit of {MAX_ARGUMENTS}",
52            py_match.arguments.len()
53        )));
54    }
55
56    // Convert event
57    let event = py_match
58        .event
59        .as_deref()
60        .map(parse_hook_event)
61        .transpose()?;
62
63    // Convert string matches
64    let tool_name = py_match
65        .tool_name
66        .as_ref()
67        .map(convert_string_match)
68        .transpose()?;
69    let cwd = py_match
70        .cwd
71        .as_ref()
72        .map(convert_string_match)
73        .transpose()?;
74    let git_branch = py_match
75        .git_branch
76        .as_ref()
77        .map(convert_string_match)
78        .transpose()?;
79
80    // Convert arguments
81    let arguments = if py_match.arguments.is_empty() {
82        None
83    } else {
84        let mut args = Vec::with_capacity(py_match.arguments.len());
85        for (name, value) in &py_match.arguments {
86            validate_pattern_length(name, "argument name")?;
87            args.push(ArgumentMatch {
88                name: name.clone(),
89                value: convert_string_match(value)?,
90            });
91        }
92        Some(args)
93    };
94
95    Ok(HookMatch {
96        event,
97        tool_name,
98        arguments,
99        cwd,
100        git_branch,
101    })
102}

104/// Convert a Python `PyStringMatch` to a Rust `StringMatch`.
105fn convert_string_match(py_sm: &PyStringMatch) -> PyResult<StringMatch> {
106    match py_sm {
107        PyStringMatch::Exact { value } => {
108            validate_pattern_length(value, "exact pattern")?;
109            Ok(StringMatch::Exact(value.clone()))
110        }
111        PyStringMatch::Prefix { value } => {
112            validate_pattern_length(value, "prefix pattern")?;
113            Ok(StringMatch::Prefix(value.clone()))
114        }
115        PyStringMatch::Suffix { value } => {
116            validate_pattern_length(value, "suffix pattern")?;
117            Ok(StringMatch::Suffix(value.clone()))
118        }
119        PyStringMatch::Contains { value } => {
120            validate_pattern_length(value, "contains pattern")?;
121            Ok(StringMatch::Contains(value.clone()))
122        }
123        PyStringMatch::Regex { pattern } => {
124            if pattern.len() > MAX_REGEX_PATTERN_LENGTH {
125                return Err(PyValueError::new_err(format!(
126                    "regex pattern length {} exceeds limit of {MAX_REGEX_PATTERN_LENGTH}",
127                    pattern.len()
128                )));
129            }
130            Ok(StringMatch::Regex(pattern.clone()))
131        }
132    }
133}
134
135/// Validate that a pattern doesn't exceed the general length limit.
136fn validate_pattern_length(value: &str, label: &str) -> PyResult<()> {
137    if value.len() > MAX_PATTERN_LENGTH {
138        return Err(PyValueError::new_err(format!(
139            "{label} length {} exceeds limit of {MAX_PATTERN_LENGTH}",
140            value.len()
141        )));
142    }
143    Ok(())
144}

146/// Parse a hook event string into a `HookEvent` enum.
147fn parse_hook_event(s: &str) -> PyResult<HookEvent> {
148    match s {
149        "PreToolUse" => Ok(HookEvent::PreToolUse),
150        "PostToolUse" => Ok(HookEvent::PostToolUse),
151        "Stop" => Ok(HookEvent::Stop),
152        "SubagentStop" => Ok(HookEvent::SubagentStop),
153        "UserPromptSubmit" => Ok(HookEvent::UserPromptSubmit),
154        "SessionStart" => Ok(HookEvent::SessionStart),
155        "SessionEnd" => Ok(HookEvent::SessionEnd),
156        "PreCompact" => Ok(HookEvent::PreCompact),
157        "Notification" => Ok(HookEvent::Notification),
158        _ => Err(PyValueError::new_err(format!(
159            "unknown hook event: {s:?}. Expected one of: PreToolUse, PostToolUse, Stop, \
160             SubagentStop, UserPromptSubmit, SessionStart, SessionEnd, PreCompact, Notification"
161        ))),
162    }
163}

169    #[test]
170    fn empty_match_rejected_without_match_all() {
171        let py_match = PyHookMatch {
172            event: None,
173            tool_name: None,
174            arguments: vec![],
175            session_id: None,
176            cwd: None,
177            git_branch: None,
178            match_all: false,
179        };

181        assert!(result.is_err());
182    }

186        let py_match = PyHookMatch {
187            event: None,
188            tool_name: None,
189            arguments: vec![],
190            session_id: None,
191            cwd: None,

193            match_all: true,
