1//! Evaluation trace types for debugging matcher behavior.
2//!
3//! Trace types mirror the runtime types ([`Predicate`](crate::Predicate),
4//! [`Matcher`](crate::Matcher)) but capture evaluation results instead of
5//! inputs. Use `evaluate_with_trace()` to get full visibility into the
6//! matcher engine's decision path.
7//!
8//! # Two Levels of Trace
9//!
10//! - [`PredicateTrace`] — Per-predicate: which sub-expressions matched?
11//! - [`EvalTrace`] — Per-matcher: which field matchers fired, what path was taken?
12//!
13//! # Example
14//!
15//! ```ignore
16//! let trace = matcher.evaluate_with_trace(&ctx);
17//! println!("Result: {:?}", trace.result);
18//! for step in &trace.steps {
19//!     println!("  field_matcher[{}]: matched={}", step.index, step.matched);
20//! }
21//! ```
22
23use std::fmt;

25/// Trace of a predicate evaluation.
26///
27/// Mirrors [`Predicate`](crate::Predicate) structure but captures results.
28///
29/// In And/Or, ALL children are evaluated (no short-circuit) for maximum
30/// debugging value. The `matched` result is still correct.
31pub enum PredicateTrace {
32    /// A single predicate evaluation.
33    Single {
34        /// Whether this predicate matched.
35        matched: bool,
36        /// Debug description of the `DataInput` (e.g., `"ToolNameInput"`).
37        input: String,
38        /// The `MatchingData` extracted from context (Debug format).
39        data: String,
40        /// Debug description of the `InputMatcher` (e.g., `"ExactMatcher(\"Bash\")"`).
41        matcher: String,
42    },
43    /// AND: all children must match.
44    And {
45        /// Whether all children matched.
46        matched: bool,
47        /// Trace of each child (all evaluated, no short-circuit).
48        children: Vec<PredicateTrace>,
49    },
50    /// OR: any child must match.
51    Or {
52        /// Whether any child matched.
53        matched: bool,
54        /// Trace of each child (all evaluated, no short-circuit).
55        children: Vec<PredicateTrace>,
56    },
57    /// NOT: inverts inner result.
58    Not {
59        /// Whether the NOT predicate matched (i.e., inner did NOT match).
60        matched: bool,
61        /// Trace of the inner predicate.
62        inner: Box<PredicateTrace>,
63    },
64}
65
66impl PredicateTrace {
67    /// Get the overall match result of this predicate.
68    #[must_use]
69    pub fn matched(&self) -> bool {
70        match self {
71            Self::Single { matched, .. }
72            | Self::And { matched, .. }
73            | Self::Or { matched, .. }
74            | Self::Not { matched, .. } => *matched,
75        }
76    }
77}
78
79impl fmt::Debug for PredicateTrace {
80    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
81        match self {
82            Self::Single {
83                matched,
84                input,
85                data,
86                matcher,
87            } => f
88                .debug_struct("Single")
89                .field("matched", matched)
90                .field("input", input)
91                .field("data", data)
92                .field("matcher", matcher)
93                .finish(),
94            Self::And { matched, children } => f
95                .debug_struct("And")
96                .field("matched", matched)
97                .field("children", children)
98                .finish(),
99            Self::Or { matched, children } => f
100                .debug_struct("Or")
101                .field("matched", matched)
102                .field("children", children)
103                .finish(),
104            Self::Not { matched, inner } => f
105                .debug_struct("Not")
106                .field("matched", matched)
107                .field("inner", inner)
108                .finish(),
109        }
110    }
111}

113/// Trace of a full [`Matcher`](crate::Matcher) evaluation.
114///
115/// Contains the same result as `evaluate()` plus the full evaluation path:
116/// which field matchers were checked, which predicates fired, and whether
117/// the fallback was used.
118///
119/// # INV: `result` == `evaluate()` result
120///
121/// The `result` field always equals what [`Matcher::evaluate()`](crate::Matcher::evaluate)
122/// would return for the same input.
123pub struct EvalTrace<A> {
124    /// The final result (identical to what `evaluate()` returns).
125    pub result: Option<A>,
126    /// Trace of each field matcher that was evaluated (in order).
127    /// Stops after the first match (preserves first-match-wins).
128    pub steps: Vec<EvalStep<A>>,
129    /// Whether the `on_no_match` fallback was used.
130    pub used_fallback: bool,
131}
132
133impl<A: fmt::Debug> fmt::Debug for EvalTrace<A> {
134    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
135        f.debug_struct("EvalTrace")
136            .field("result", &self.result)
137            .field("steps", &self.steps)
138            .field("used_fallback", &self.used_fallback)
139            .finish()
140    }
141}
142
143/// One field matcher's evaluation in a trace.
144pub struct EvalStep<A> {
145    /// Index in `matcher_list` (0-based).
146    pub index: usize,
147    /// Did the predicate match?
148    pub matched: bool,
149    /// Full predicate evaluation trace.
150    pub predicate_trace: PredicateTrace,
151    /// If predicate matched, what happened with `OnMatch`?
152    pub on_match: Option<OnMatchTrace<A>>,
153}
154
155impl<A: fmt::Debug> fmt::Debug for EvalStep<A> {
156    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
157        f.debug_struct("EvalStep")
158            .field("index", &self.index)
159            .field("matched", &self.matched)
160            .field("predicate_trace", &self.predicate_trace)
161            .field("on_match", &self.on_match)
162            .finish()
163    }
164}
165
166/// What happened when a matched predicate resolved its `OnMatch`.
167pub enum OnMatchTrace<A> {
168    /// Returned an action directly.
169    Action(A),
170    /// Delegated to a nested matcher (recursive trace).
171    Nested(Box<EvalTrace<A>>),
172}
173
174impl<A: fmt::Debug> fmt::Debug for OnMatchTrace<A> {
175    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
176        match self {
177            Self::Action(a) => f.debug_tuple("Action").field(a).finish(),
178            Self::Nested(t) => f.debug_tuple("Nested").field(t).finish(),
179        }
180    }
181}

183#[cfg(test)]
184mod tests {
185    use super::*;
186
187    #[test]
188    fn predicate_trace_matched_single() {
189        let trace = PredicateTrace::Single {
190            matched: true,
191            input: "TestInput".into(),
192            data: "String(\"hello\")".into(),
193            matcher: "ExactMatcher(\"hello\")".into(),
194        };
195        assert!(trace.matched());
196    }

198    #[test]
199    fn predicate_trace_matched_and() {
200        let trace = PredicateTrace::And {
201            matched: false,
202            children: vec![],
203        };
204        assert!(!trace.matched());
205    }
206
207    #[test]
208    fn predicate_trace_matched_or() {
209        let trace = PredicateTrace::Or {
210            matched: true,
211            children: vec![],
212        };
213        assert!(trace.matched());
214    }
215
216    #[test]
217    fn predicate_trace_matched_not() {
218        let trace = PredicateTrace::Not {
219            matched: true,
220            inner: Box::new(PredicateTrace::Single {
221                matched: false,
222                input: String::new(),
223                data: String::new(),
224                matcher: String::new(),
225            }),
226        };
227        assert!(trace.matched());
228    }

230    #[test]
