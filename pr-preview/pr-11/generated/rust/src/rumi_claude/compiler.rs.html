1//! Compiler: `HookMatch` -> rumi `Matcher<HookContext, A>`
2//!
3//! Translates user-friendly hook match configuration into efficient
4//! runtime matchers operating on [`HookContext`].
5
6use crate::config::{ArgumentMatch, CompileError, HookMatch, StringMatch};
7use crate::context::{HookContext, HookEvent};
8use crate::inputs::{ArgumentInput, CwdInput, EventInput, GitBranchInput, ToolNameInput};
9use rumi::prelude::*;
10
11/// Extension trait for compiling `HookMatch` to rumi `Matcher`.
12pub trait HookMatchExt {
13    /// Compile this `HookMatch` into a rumi `Matcher`.
14    ///
15    /// The resulting matcher operates on [`HookContext`] and returns
16    /// the provided action when all conditions match.
17    ///
18    /// # Errors
19    ///
20    /// Returns [`CompileError`] if any regex pattern is invalid.
21    fn compile<A: Clone + Send + Sync + 'static>(
22        &self,
23        action: A,
24    ) -> Result<Matcher<HookContext, A>, CompileError>;
25
26    /// Compile this `HookMatch` into a `Predicate` (without action).
27    ///
28    /// # Errors
29    ///
30    /// Returns [`CompileError`] if any regex pattern is invalid.
31    fn to_predicate(&self) -> Result<Predicate<HookContext>, CompileError>;
32}
33
34impl HookMatchExt for HookMatch {
35    fn compile<A: Clone + Send + Sync + 'static>(
36        &self,
37        action: A,
38    ) -> Result<Matcher<HookContext, A>, CompileError> {
39        let predicate = self.to_predicate()?;
40        Ok(Matcher::new(
41            vec![FieldMatcher::new(predicate, OnMatch::Action(action))],
42            None,
43        ))
44    }
45
46    fn to_predicate(&self) -> Result<Predicate<HookContext>, CompileError> {
47        let mut predicates: Vec<Predicate<HookContext>> = Vec::new();
48
49        // Event matching (exact only — event names are a known enum)
50        if let Some(event) = &self.event {
51            predicates.push(compile_event_match(*event));
52        }
53
54        // Tool name matching
55        if let Some(tool_match) = &self.tool_name {
56            predicates.push(compile_string_predicate(
57                Box::new(ToolNameInput),
58                tool_match,
59            )?);
60        }
61
62        // Argument matching (all ANDed)
63        if let Some(arguments) = &self.arguments {
64            for arg_match in arguments {
65                predicates.push(compile_argument_match(arg_match)?);
66            }
67        }
68
69        // CWD matching
70        if let Some(cwd_match) = &self.cwd {
71            predicates.push(compile_string_predicate(Box::new(CwdInput), cwd_match)?);
72        }
73
74        // Git branch matching
75        if let Some(branch_match) = &self.git_branch {
76            predicates.push(compile_string_predicate(
77                Box::new(GitBranchInput),
78                branch_match,
79            )?);
80        }
81
82        // Empty match = match everything (same as rumi-http)
83        if predicates.is_empty() {
84            return Ok(Predicate::Single(SinglePredicate::new(
85                Box::new(EventInput),
86                Box::new(PrefixMatcher::new("")), // matches any event string
87            )));
88        }
89
90        if predicates.len() == 1 {
91            Ok(predicates.pop().unwrap())
92        } else {
93            Ok(Predicate::And(predicates))
94        }
95    }
96}

98/// Compile an event match to a predicate (always exact).
99fn compile_event_match(event: HookEvent) -> Predicate<HookContext> {
100    Predicate::Single(SinglePredicate::new(
101        Box::new(EventInput),
102        Box::new(ExactMatcher::new(event.as_str())),
103    ))
104}

106/// Compile a `StringMatch` + `DataInput` into a predicate.
107fn compile_string_predicate(
108    input: Box<dyn DataInput<HookContext>>,
109    string_match: &StringMatch,
110) -> Result<Predicate<HookContext>, CompileError> {
111    let matcher = string_match.to_input_matcher()?;
112    Ok(Predicate::Single(SinglePredicate::new(input, matcher)))
113}
114
115/// Compile an argument match to a predicate.
116fn compile_argument_match(
117    arg_match: &ArgumentMatch,
118) -> Result<Predicate<HookContext>, CompileError> {
119    let input = Box::new(ArgumentInput::new(&arg_match.name));
120    let matcher = arg_match.value.to_input_matcher()?;
121    Ok(Predicate::Single(SinglePredicate::new(input, matcher)))
122}

124/// Compile multiple `HookMatch` entries into a single `Matcher`.
125///
126/// Multiple matches are `ORed` together: the first matching rule wins.
127///
128/// # Errors
129///
130/// Returns [`CompileError`] if any regex pattern is invalid.
131pub fn compile_hook_matches<A: Clone + Send + Sync + 'static>(
132    matches: &[HookMatch],
133    action: A,
134    on_no_match: Option<A>,
135) -> Result<Matcher<HookContext, A>, CompileError> {
136    if matches.is_empty() {
137        // Empty matches = match everything
138        return Ok(Matcher::new(
139            vec![FieldMatcher::new(
140                Predicate::Single(SinglePredicate::new(
141                    Box::new(EventInput),
142                    Box::new(PrefixMatcher::new("")),
143                )),
144                OnMatch::Action(action),
145            )],
146            on_no_match.map(OnMatch::Action),
147        ));
148    }
149
150    if matches.len() == 1 {
151        let predicate = matches[0].to_predicate()?;
152        return Ok(Matcher::new(
153            vec![FieldMatcher::new(predicate, OnMatch::Action(action))],
154            on_no_match.map(OnMatch::Action),
155        ));
156    }
157
158    // Multiple matches: OR them together
159    let predicates: Vec<Predicate<HookContext>> = matches
160        .iter()
161        .map(HookMatchExt::to_predicate)
162        .collect::<Result<_, _>>()?;
163
164    Ok(Matcher::new(
165        vec![FieldMatcher::new(
166            Predicate::Or(predicates),
167            OnMatch::Action(action),
168        )],
169        on_no_match.map(OnMatch::Action),
170    ))
171}

177/// Result of tracing a `HookMatch` against a `HookContext`.
178///
179/// Shows which fields matched and which didn't, with expected vs actual values.
180#[derive(Debug)]
181pub struct HookMatchTrace {
182    /// Whether the overall match succeeded (all steps must match).
183    pub matched: bool,
184    /// Per-field trace steps.
185    pub steps: Vec<TraceStep>,
186}
187
188/// One field's evaluation result in a trace.
189#[derive(Debug)]
190pub struct TraceStep {
191    /// Field name (e.g., "event", "`tool_name`", "argument[command]").
192    pub field: String,
193    /// What was expected (e.g., `Exact("Bash")`).
194    pub expected: String,
195    /// What was found in the context.
196    pub actual: String,
197    /// Whether this individual step matched.
198    pub matched: bool,
199}
200
201impl HookMatch {
202    /// Trace this match against a context, reporting per-field results.
203    ///
204    /// Use this to debug "why was this hook triggered?" or "why didn't it match?"
205    #[must_use]
206    pub fn trace(&self, ctx: &HookContext) -> HookMatchTrace {
207        let mut steps = Vec::new();
208
209        if let Some(event) = &self.event {
210            let actual = ctx.event().as_str().to_string();
211            let expected_str = event.as_str().to_string();
212            let matched = actual == expected_str;
213            steps.push(TraceStep {
214                field: "event".into(),
215                expected: expected_str,
216                actual,
217                matched,
218            });
219        }
220
221        if let Some(tool_match) = &self.tool_name {
222            let actual = ctx.tool_name().to_string();
223            let matched = trace_string_match(tool_match, &actual);
224            steps.push(TraceStep {
225                field: "tool_name".into(),
226                expected: tool_match.to_string(),
227                actual,
228                matched,
229            });
230        }
231
232        if let Some(arguments) = &self.arguments {
233            for arg_match in arguments {
234                let actual = ctx
235                    .argument(&arg_match.name)
236                    .unwrap_or("<missing>")
237                    .to_string();
238                let matched = ctx
239                    .argument(&arg_match.name)
240                    .is_some_and(|v| trace_string_match(&arg_match.value, v));
241                steps.push(TraceStep {
242                    field: format!("argument[{}]", arg_match.name),
243                    expected: arg_match.value.to_string(),
244                    actual,
245                    matched,
246                });
247            }
248        }
249
250        if let Some(cwd_match) = &self.cwd {
251            let actual = ctx.cwd().to_string();
252            let matched = trace_string_match(cwd_match, &actual);
253            steps.push(TraceStep {
254                field: "cwd".into(),
255                expected: cwd_match.to_string(),
256                actual,
257                matched,
258            });
259        }
260
261        if let Some(branch_match) = &self.git_branch {
262            let actual = ctx.git_branch().unwrap_or("<none>").to_string();
263            let matched = ctx
264                .git_branch()
265                .is_some_and(|v| trace_string_match(branch_match, v));
266            steps.push(TraceStep {
267                field: "git_branch".into(),
268                expected: branch_match.to_string(),
269                actual,
270                matched,
271            });
272        }
273
274        let overall = steps.iter().all(|s| s.matched);
275        HookMatchTrace {
276            matched: overall,
277            steps,
278        }
279    }
280}

282/// Evaluate a `StringMatch` against a concrete value (for trace only).
283///
284/// Compiles the `StringMatch` to an `InputMatcher` and checks against
285/// `MatchingData::String`. Returns false if the regex is invalid.
286fn trace_string_match(string_match: &StringMatch, value: &str) -> bool {
287    string_match
288        .to_input_matcher()
289        .map(|m| m.matches(&MatchingData::String(value.to_string())))
290        .unwrap_or(false)
291}
