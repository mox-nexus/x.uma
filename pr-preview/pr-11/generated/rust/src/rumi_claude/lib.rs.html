1//! rumi-claude: Claude Code hooks domain
2//!
3//! Provides context types, `DataInput` extractors, and a **domain compiler**
4//! for matching Claude Code hook events.
5//!
6//! # Compiler (recommended)
7//!
8//! ```ignore
9//! use rumi_claude::prelude::*;
10//!
11//! // Declarative: block dangerous Bash commands
12//! let rule = HookMatch {
13//!     event: Some(HookEvent::PreToolUse),
14//!     tool_name: Some(StringMatch::Exact("Bash".into())),
15//!     arguments: Some(vec![ArgumentMatch {
16//!         name: "command".into(),
17//!         value: StringMatch::Contains("rm -rf".into()),
18//!     }]),
19//!     ..Default::default()
20//! };
21//! let matcher = rule.compile("block")?;
22//!
23//! let ctx = HookContext::pre_tool_use("Bash")
24//!     .with_arg("command", "rm -rf /important");
25//! assert_eq!(matcher.evaluate(&ctx), Some("block"));
26//! ```
27//!
28//! # Trace (debugging)
29//!
30//! ```ignore
31//! let trace = rule.trace(&ctx);
32//! for step in &trace.steps {
33//!     println!("{}: expected={}, actual={}, matched={}",
34//!         step.field, step.expected, step.actual, step.matched);
35//! }
36//! ```

58#[cfg(test)]
59mod tests {
60    use super::*;
61    use rumi::prelude::*;
62
63    #[test]
64    fn test_hook_context_builder() {
65        let ctx = HookContext::pre_tool_use("Bash").with_arg("command", "ls -la");
66
67        assert_eq!(ctx.event(), HookEvent::PreToolUse);
68        assert_eq!(ctx.tool_name(), "Bash");
69        assert_eq!(ctx.argument("command"), Some("ls -la"));
70    }
71
72    #[test]
73    fn test_tool_name_input() {
74        let ctx = HookContext::pre_tool_use("Write");
75        assert_eq!(
76            ToolNameInput.get(&ctx),
77            MatchingData::String("Write".into())
78        );
79    }
80
81    #[test]
82    fn test_event_input() {
83        let ctx = HookContext::post_tool_use("Read");
84        assert_eq!(
85            EventInput.get(&ctx),
86            MatchingData::String("PostToolUse".into())
87        );
88    }
89
90    #[test]
91    fn test_argument_input() {
92        let ctx = HookContext::pre_tool_use("Bash").with_arg("command", "echo hello");
93
94        let input = ArgumentInput::new("command");
95        assert_eq!(input.get(&ctx), MatchingData::String("echo hello".into()));
96    }
97
98    #[test]
99    fn test_dangerous_command_matcher() {
100        let ctx = HookContext::pre_tool_use("Bash").with_arg("command", "rm -rf /important");
101
102        // Match dangerous Bash commands
103        let matcher: Matcher<HookContext, &str> = Matcher::new(
104            vec![FieldMatcher::new(
105                Predicate::And(vec![
106                    Predicate::Single(SinglePredicate::new(
107                        Box::new(ToolNameInput),
108                        Box::new(ExactMatcher::new("Bash")),
109                    )),
110                    Predicate::Single(SinglePredicate::new(
111                        Box::new(ArgumentInput::new("command")),
112                        Box::new(ContainsMatcher::new("rm -rf")),
113                    )),
114                ]),
115                OnMatch::Action("block"),
116            )],
117            Some(OnMatch::Action("allow")),
118        );
119
120        assert_eq!(matcher.evaluate(&ctx), Some("block"));
121    }
