1//! Config-driven HTTP matcher exposed to Python.
2//!
3//! Takes a JSON config string, compiles it via Rust registry, and evaluates
4//! against HTTP request contexts. Same `MatcherConfig<String>` format used
5//! by all implementations.
6
7use std::collections::HashMap;
8
9use pyo3::exceptions::PyValueError;
10use pyo3::prelude::*;
11use rumi::prelude::*;
12use rumi_http::HttpRequest;
13
14/// An opaque compiled HTTP matcher.
15///
16/// Created via `HttpMatcher.from_config()`, immutable after construction.
17/// Evaluates HTTP requests against compiled matcher trees.
18///
19/// # Thread Safety
20///
21/// `HttpMatcher` is immutable and safe to share across threads.
22#[pyclass(frozen)]
23pub struct HttpMatcher {
24    inner: Matcher<HttpRequest, String>,
25}
26
27#[pymethods]
28impl HttpMatcher {
29    /// Load a matcher from a JSON config string.
30    ///
31    /// The config format is `MatcherConfig<String>` — the same JSON shape used
32    /// by all x.uma implementations (rumi, puma, bumi).
33    ///
34    /// # Supported input type URLs
35    ///
36    /// - `xuma.http.v1.PathInput` — request path (config: `{}`)
37    /// - `xuma.http.v1.MethodInput` — HTTP method (config: `{}`)
38    /// - `xuma.http.v1.HeaderInput` — header value (config: `{"name": "..."}`)
39    /// - `xuma.http.v1.QueryParamInput` — query parameter (config: `{"name": "..."}`)
40    ///
41    /// # Errors
42    ///
43    /// Raises `ValueError` if:
44    /// - JSON config is malformed
45    /// - Unknown type URL (error lists available URLs)
46    /// - Invalid regex pattern
47    /// - Depth/width limits exceeded
48    #[staticmethod]
49    fn from_config(json_config: &str) -> PyResult<Self> {
50        let config: rumi::MatcherConfig<String> = serde_json::from_str(json_config)
51            .map_err(|e| PyValueError::new_err(format!("invalid config JSON: {e}")))?;
52
53        let registry = build_http_registry();
54        let matcher = registry
55            .load_matcher(config)
56            .map_err(|e| PyValueError::new_err(e.to_string()))?;
57
58        matcher
59            .validate()
60            .map_err(|e| PyValueError::new_err(e.to_string()))?;
61
62        Ok(Self { inner: matcher })
63    }
64
65    /// Evaluate an HTTP request against compiled matcher rules.
66    ///
67    /// # Arguments
68    ///
69    /// * `method` — HTTP method (e.g., "GET", "POST").
70    /// * `path` — Request path (e.g., "/api/users").
71    /// * `headers` — Request headers as key-value pairs (keys are case-insensitive).
72    /// * `query_params` — Query parameters as key-value pairs.
73    ///
74    /// # Returns
75    ///
76    /// The action string if the request matched, or `None`.
77    #[pyo3(signature = (method, path, headers = None, query_params = None))]
78    fn evaluate(
79        &self,
80        method: &str,
81        path: &str,
82        headers: Option<HashMap<String, String>>,
83        query_params: Option<HashMap<String, String>>,
84    ) -> Option<String> {
85        let req = build_request(method, path, headers, query_params);
86        self.inner.evaluate(&req)
87    }
88
89    /// Trace evaluation for debugging.
90    ///
91    /// Returns the same result as `evaluate()` plus a detailed trace.
92    #[pyo3(signature = (method, path, headers = None, query_params = None))]
93    fn trace(
94        &self,
95        method: &str,
96        path: &str,
97        headers: Option<HashMap<String, String>>,
98        query_params: Option<HashMap<String, String>>,
99    ) -> super::matcher::PyTraceResult {
100        let req = build_request(method, path, headers, query_params);
101        let trace = self.inner.evaluate_with_trace(&req);
102
103        let steps: Vec<super::matcher::PyTraceStep> = trace
104            .steps
105            .iter()
106            .map(|step| super::matcher::PyTraceStep {
107                index: step.index,
108                matched: step.matched,
109                predicate: format!("{:?}", step.predicate_trace),
110            })
111            .collect();
112
113        super::matcher::PyTraceResult {
114            result: trace.result,
115            steps,
116            used_fallback: trace.used_fallback,
117        }
118    }
119
120    #[allow(clippy::unused_self)]
121    fn __repr__(&self) -> String {
122        "HttpMatcher(<compiled>)".to_string()
123    }
124}
125
126/// Build the HTTP registry for `HttpRequest`.
127fn build_http_registry() -> rumi::Registry<HttpRequest> {
128    rumi_http::register_simple(rumi::RegistryBuilder::new()).build()
129}
130
131/// Build an `HttpRequest` from Python arguments.
132fn build_request(
133    method: &str,
134    path: &str,
135    headers: Option<HashMap<String, String>>,
136    query_params: Option<HashMap<String, String>>,
137) -> HttpRequest {
138    let mut builder = HttpRequest::builder().method(method).path(path);
139
140    if let Some(hdrs) = headers {
141        for (k, v) in hdrs {
142            builder = builder.header(k, v);
143        }
144    }
145
146    if let Some(params) = query_params {
147        for (k, v) in params {
148            builder = builder.query_param(k, v);
149        }
150    }
151
152    builder.build()
153}
