1//! `StringMatch` factory functions exposed to TypeScript.
2//!
3//! Factory methods return plain JS objects with a `type` discriminant — the
4//! natural TypeScript pattern (discriminated unions). No opaque Rust struct
5//! needed for config types; only `HookMatcher` is opaque.
6//!
7//! ```ts
8//! // Bare string → exact match
9//! { toolName: "Bash" }
10//!
11//! // Factory methods
12//! { toolName: StringMatch.prefix("mcp__") }
13//! { toolName: StringMatch.regex("^(Write|Edit)$") }
14//! ```
15
16use js_sys::JsString;
17use wasm_bindgen::prelude::*;
18
19/// Factory methods for string matching configuration.
20///
21/// Each method returns a plain JS object with a `type` discriminant:
22/// - `StringMatch.exact("Bash")` → `{ type: "exact", value: "Bash" }`
23/// - `StringMatch.prefix("mcp__")` → `{ type: "prefix", value: "mcp__" }`
24/// - `StringMatch.regex("^Bash$")` → `{ type: "regex", pattern: "^Bash$" }`
25///
26/// Bare strings passed directly to `HookMatcher.compile()` fields are also
27/// treated as exact matches (Ace convenience).
28#[wasm_bindgen]
29pub struct StringMatch {
30    // Zero-size: exists only as a namespace for static methods.
31    _private: u8,
32}
33
34/// Build a `{ type, value }` JS object.
35fn make_match(kind: &str, key: &str, val: &str) -> JsValue {
36    let obj = js_sys::Object::new();
37    js_sys::Reflect::set(&obj, &JsString::from("type"), &JsString::from(kind)).unwrap();
38    js_sys::Reflect::set(&obj, &JsString::from(key), &JsString::from(val)).unwrap();
39    obj.into()
40}
41
42#[wasm_bindgen]
43#[allow(clippy::needless_pass_by_value)] // wasm-bindgen requires owned String across FFI
44impl StringMatch {
45    /// Exact equality: `{ type: "exact", value }`.
46    pub fn exact(value: String) -> JsValue {
47        make_match("exact", "value", &value)
48    }
49
50    /// Starts with prefix: `{ type: "prefix", value }`.
51    pub fn prefix(value: String) -> JsValue {
52        make_match("prefix", "value", &value)
53    }
54
55    /// Ends with suffix: `{ type: "suffix", value }`.
56    pub fn suffix(value: String) -> JsValue {
57        make_match("suffix", "value", &value)
58    }
59
60    /// Contains substring: `{ type: "contains", value }`.
61    pub fn contains(value: String) -> JsValue {
62        make_match("contains", "value", &value)
63    }
64
65    /// Matches regex (Rust `regex` crate — guaranteed linear time): `{ type: "regex", pattern }`.
66    pub fn regex(pattern: String) -> JsValue {
67        make_match("regex", "pattern", &pattern)
68    }
69}
