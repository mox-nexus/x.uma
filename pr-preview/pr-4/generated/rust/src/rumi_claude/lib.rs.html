1//! rumi-claude: Claude Code hooks domain
2//!
3//! Provides context and `DataInput` implementations for Claude Code hook matching.
4//!
5//! # Example
6//!
7//! ```ignore
8//! use rumi_claude::prelude::*;
9//!
10//! let ctx = HookContext::pre_tool_use("Bash")
11//!     .with_arg("command", "rm -rf /");
12//!
13//! let input = ToolNameInput;
14//! assert_eq!(input.get(&ctx), MatchingData::String("Bash".into()));
15//! ```
16
17use rumi::prelude::*;
18use std::collections::HashMap;
19
20/// Hook event types in Claude Code.
21#[derive(Debug, Clone, Copy, PartialEq, Eq)]
22pub enum HookEvent {
23    PreToolUse,
24    PostToolUse,
25    Stop,
26    SubagentStop,
27}
28
29impl HookEvent {
30    /// Convert to string representation.
31    #[must_use]
32    pub fn as_str(&self) -> &'static str {
33        match self {
34            Self::PreToolUse => "PreToolUse",
35            Self::PostToolUse => "PostToolUse",
36            Self::Stop => "Stop",
37            Self::SubagentStop => "SubagentStop",
38        }
39    }
40}
41
42/// Claude Code hook context for matching.
43#[derive(Debug, Clone)]
44pub struct HookContext {
45    event: HookEvent,
46    tool_name: String,
47    arguments: HashMap<String, String>,
48}
49
50impl HookContext {
51    /// Create a PreToolUse hook context.
52    #[must_use]
53    pub fn pre_tool_use(tool_name: impl Into<String>) -> Self {
54        Self {
55            event: HookEvent::PreToolUse,
56            tool_name: tool_name.into(),
57            arguments: HashMap::new(),
58        }
59    }
60
61    /// Create a PostToolUse hook context.
62    #[must_use]
63    pub fn post_tool_use(tool_name: impl Into<String>) -> Self {
64        Self {
65            event: HookEvent::PostToolUse,
66            tool_name: tool_name.into(),
67            arguments: HashMap::new(),
68        }
69    }
70
71    /// Add an argument (builder pattern).
72    #[must_use]
73    pub fn with_arg(mut self, name: impl Into<String>, value: impl Into<String>) -> Self {
74        self.arguments.insert(name.into(), value.into());
75        self
76    }
77
78    /// Get the hook event type.
79    #[must_use]
80    pub fn event(&self) -> HookEvent {
81        self.event
82    }
83
84    /// Get the tool name.
85    #[must_use]
86    pub fn tool_name(&self) -> &str {
87        &self.tool_name
88    }
89
90    /// Get an argument by name.
91    #[must_use]
92    pub fn argument(&self, name: &str) -> Option<&str> {
93        self.arguments.get(name).map(String::as_str)
94    }
95}
96
97/// Extracts the hook event type.
98#[derive(Debug, Clone)]
99pub struct EventInput;
100
101impl DataInput<HookContext> for EventInput {
102    fn get(&self, ctx: &HookContext) -> MatchingData {
103        MatchingData::String(ctx.event.as_str().to_string())
104    }
105}
106
107/// Extracts the tool name.
108#[derive(Debug, Clone)]
109pub struct ToolNameInput;
110
111impl DataInput<HookContext> for ToolNameInput {
112    fn get(&self, ctx: &HookContext) -> MatchingData {
113        MatchingData::String(ctx.tool_name.clone())
114    }
115}
116
117/// Extracts a tool argument by name.
118#[derive(Debug, Clone)]
119pub struct ArgumentInput {
120    name: String,
121}
122
123impl ArgumentInput {
124    /// Create a new argument input extractor.
125    pub fn new(name: impl Into<String>) -> Self {
126        Self { name: name.into() }
127    }
128}
129
130impl DataInput<HookContext> for ArgumentInput {
131    fn get(&self, ctx: &HookContext) -> MatchingData {
132        ctx.argument(&self.name)
133            .map(|s| MatchingData::String(s.to_string()))
134            .unwrap_or(MatchingData::None)
135    }
136}

144#[cfg(test)]
145mod tests {
146    use super::*;
147
148    #[test]
149    fn test_hook_context_builder() {
150        let ctx = HookContext::pre_tool_use("Bash")
151            .with_arg("command", "ls -la");
152
153        assert_eq!(ctx.event(), HookEvent::PreToolUse);
154        assert_eq!(ctx.tool_name(), "Bash");
155        assert_eq!(ctx.argument("command"), Some("ls -la"));
156    }
157
158    #[test]
159    fn test_tool_name_input() {
160        let ctx = HookContext::pre_tool_use("Write");
161        assert_eq!(ToolNameInput.get(&ctx), MatchingData::String("Write".into()));
162    }
163
164    #[test]
165    fn test_event_input() {
166        let ctx = HookContext::post_tool_use("Read");
167        assert_eq!(EventInput.get(&ctx), MatchingData::String("PostToolUse".into()));
168    }
169
170    #[test]
171    fn test_argument_input() {
172        let ctx = HookContext::pre_tool_use("Bash")
173            .with_arg("command", "echo hello");
174
175        let input = ArgumentInput::new("command");
176        assert_eq!(input.get(&ctx), MatchingData::String("echo hello".into()));
177    }
178
179    #[test]
180    fn test_dangerous_command_matcher() {
181        let ctx = HookContext::pre_tool_use("Bash")
182            .with_arg("command", "rm -rf /important");
183
184        // Match dangerous Bash commands
185        let matcher: Matcher<HookContext, &str> = Matcher::new(
186            vec![
187                FieldMatcher::new(
188                    Predicate::And(vec![
189                        Predicate::Single(SinglePredicate::new(
190                            Box::new(ToolNameInput),
191                            Box::new(ExactMatcher::new("Bash")),
192                        )),
193                        Predicate::Single(SinglePredicate::new(
194                            Box::new(ArgumentInput::new("command")),
195                            Box::new(ContainsMatcher::new("rm -rf")),
196                        )),
197                    ]),
198                    OnMatch::Action("block"),
199                ),
200            ],
201            Some(OnMatch::Action("allow")),
202        );
203
204        assert_eq!(matcher.evaluate(&ctx), Some("block"));
205    }
