1//! Opaque compiled matchers exposed to TypeScript.
2//!
3//! The key insight: don't expose the Rust type tree across FFI.
4//! Config in → compile in Rust → evaluate in Rust → simple types out.
5
6use rumi::prelude::*;
7use rumi_claude::{compile_hook_matches, HookContext, HookEvent};
8use wasm_bindgen::prelude::*;
9
10use crate::convert;

12/// An opaque compiled hook matcher.
13///
14/// Created via `HookMatcher.compile()`, immutable after construction.
15/// Evaluates Claude Code hook contexts against compiled rules.
16#[wasm_bindgen]
17pub struct HookMatcher {
18    inner: Matcher<HookContext, String>,
19}
20
21#[wasm_bindgen]
22#[allow(clippy::needless_pass_by_value)] // wasm-bindgen requires owned JsValue across FFI
23impl HookMatcher {
24    /// Compile hook match rules into an immutable matcher.
25    ///
26    /// # Arguments
27    ///
28    /// * `rules` — Array of plain objects (`ORed` together). Each object may have:
29    ///   - `event?: string` — Hook event type
30    ///   - `toolName?: string | StringMatch` — Tool name (bare string = exact match)
31    ///   - `arguments?: Array<[string, string | StringMatch]>` — Argument matchers
32    ///   - `cwd?: string | StringMatch` — Working directory
33    ///   - `gitBranch?: string | StringMatch` — Git branch
34    ///   - `matchAll?: boolean` — Allow empty catch-all (fail-closed guard)
35    /// * `action` — Action to return when any rule matches.
36    /// * `fallback` — Action to return when no rule matches (default: `undefined`).
37    ///
38    /// # Errors
39    ///
40    /// Throws if:
41    /// - Any rule has an invalid regex pattern
42    /// - An empty rule doesn't have `matchAll: true`
43    /// - Input limits exceeded (pattern length, argument count, rule count)
44    /// - Compiled matcher exceeds depth limit
45    pub fn compile(
46        rules: JsValue,
47        action: String,
48        fallback: Option<String>,
49    ) -> Result<HookMatcher, JsValue> {
50        let rules_array = js_sys::Array::from(&rules);
51
52        // Validate rule count
53        if rules_array.length() > convert::MAX_RULES {
54            return Err(JsValue::from_str(&format!(
55                "too many rules: {} exceeds limit of {}",
56                rules_array.length(),
57                convert::MAX_RULES,
58            )));
59        }
60
61        // Convert JS objects to Rust HookMatch
62        let mut rust_rules = Vec::with_capacity(rules_array.length() as usize);
63        for rule in rules_array.iter() {
64            rust_rules.push(convert::convert_hook_match_from_js(&rule)?);
65        }
66
67        // Compile via rumi-claude compiler
68        let matcher = compile_hook_matches(&rust_rules, action, fallback)
69            .map_err(|e| JsValue::from_str(&e.to_string()))?;
70
71        // Validate depth (Vector: non-negotiable)
72        matcher
73            .validate()
74            .map_err(|e| JsValue::from_str(&e.to_string()))?;
75
76        Ok(Self { inner: matcher })
77    }
78
79    /// Evaluate a hook context against compiled rules.
80    ///
81    /// Accepts a plain object with camelCase fields:
82    /// ```js
83    /// matcher.evaluate({
84    ///   event: "PreToolUse",
85    ///   toolName: "Bash",
86    ///   arguments: { command: "rm -rf /" },
87    ///   cwd: "/home/user",
88    ///   sessionId: "abc123",
89    ///   gitBranch: "main",
90    /// })
91    /// ```
92    ///
93    /// Returns the action string if rules matched, or `undefined`.
94    pub fn evaluate(&self, context: JsValue) -> Result<Option<String>, JsValue> {
95        let ctx = build_context_from_js(&context)?;
96        Ok(self.inner.evaluate(&ctx))
97    }
98
99    /// Trace evaluation for debugging (opt-in, per Vector recommendation).
100    ///
101    /// Returns the same result as `evaluate()` plus a detailed trace
102    /// showing which rules matched and why.
103    pub fn trace(&self, context: JsValue) -> Result<JsValue, JsValue> {
104        let ctx = build_context_from_js(&context)?;
105        let trace = self.inner.evaluate_with_trace(&ctx);
106
107        let steps: Vec<TraceStepSerde> = trace
108            .steps
109            .iter()
110            .map(|step| TraceStepSerde {
111                index: step.index,
112                matched: step.matched,
113                predicate: format!("{:?}", step.predicate_trace),
114            })
115            .collect();
116
117        let result = TraceResultSerde {
118            result: trace.result,
119            steps,
120            used_fallback: trace.used_fallback,
121        };
122
123        serde_wasm_bindgen::to_value(&result).map_err(|e| JsValue::from_str(&e.to_string()))
124    }
125}

127/// Build a `HookContext` from a JS options object.
128fn build_context_from_js(val: &JsValue) -> Result<HookContext, JsValue> {
129    let get =
130        |key| js_sys::Reflect::get(val, &JsValue::from_str(key)).unwrap_or(JsValue::UNDEFINED);
131
132    // Parse event (required)
133    let event_val = get("event");
134    let event_str = event_val
135        .as_string()
136        .ok_or_else(|| JsValue::from_str("event is required and must be a string"))?;
137    let hook_event = convert::parse_hook_event(&event_str)?;
138
139    // Tool name (optional, default "")
140    let tool_name_val = get("toolName");
141    let tool_name = tool_name_val.as_string().unwrap_or_default();
142
143    // Build context
144    let mut ctx = match hook_event {
145        HookEvent::PreToolUse => HookContext::pre_tool_use(&tool_name),
146        HookEvent::PostToolUse => HookContext::post_tool_use(&tool_name),
147        HookEvent::Stop => HookContext::stop(),
148        HookEvent::SubagentStop => HookContext::subagent_stop(),
149        HookEvent::UserPromptSubmit => HookContext::user_prompt_submit(),
150        HookEvent::SessionStart => HookContext::session_start(),
151        HookEvent::SessionEnd => HookContext::session_end(),
152        HookEvent::PreCompact => HookContext::pre_compact(),
153        HookEvent::Notification => HookContext::notification(),
154    };
155
156    // Arguments (optional, Record<string, string>)
157    let args_val = get("arguments");
158    if !args_val.is_undefined() && !args_val.is_null() {
159        let entries = js_sys::Object::entries(&js_sys::Object::from(args_val));
160        for entry in entries.iter() {
161            let pair = js_sys::Array::from(&entry);
162            let key = pair
163                .get(0)
164                .as_string()
165                .ok_or_else(|| JsValue::from_str("argument key must be a string"))?;
166            let value = pair
167                .get(1)
168                .as_string()
169                .ok_or_else(|| JsValue::from_str("argument value must be a string"))?;
170            ctx = ctx.with_arg(key, value);
171        }
172    }
173
174    // Preserve session_id="" vs git_branch=None semantics (Dijkstra)
175    let sid_val = get("sessionId");
176    if let Some(sid) = sid_val.as_string() {
177        ctx = ctx.with_session_id(&sid);
178    }
179
180    let cwd_val = get("cwd");
181    if let Some(cwd) = cwd_val.as_string() {
182        ctx = ctx.with_cwd(&cwd);
183    }
184
185    let branch_val = get("gitBranch");
186    if let Some(branch) = branch_val.as_string() {
187        ctx = ctx.with_git_branch(&branch);
188    }
189
190    Ok(ctx)
191}

193/// Serde-serializable trace result (for serde-wasm-bindgen output).
194#[derive(serde::Serialize)]
195#[serde(rename_all = "camelCase")]
196struct TraceResultSerde {
197    result: Option<String>,
198    steps: Vec<TraceStepSerde>,
199    used_fallback: bool,
200}
201
202/// One step in the evaluation trace.
203#[derive(serde::Serialize)]
204struct TraceStepSerde {
205    index: usize,
206    matched: bool,
207    predicate: String,
208}
