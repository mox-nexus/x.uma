1//! Config conformance fixture runner.
2//!
3//! Loads YAML fixtures that use the **registry config format** (the same JSON/YAML
4//! shape as `MatcherConfig<String>`). This tests the config-driven loading path:
5//! YAML → `MatcherConfig<String>` → `Registry::load_matcher()` → evaluate.
6//!
7//! Unlike the existing `fixture` module (which uses a custom YAML format and manually
8//! builds matchers), this module exercises the production config loading pipeline.
9
10use serde::Deserialize;
11use std::collections::HashMap;
12
13/// A config conformance test fixture.
14///
15/// The `config` field is the raw YAML/JSON value that gets deserialized as
16/// `MatcherConfig<String>` — the exact same format all implementations must support.
17#[derive(Debug, Deserialize)]
18pub struct ConfigFixture {
19    pub name: String,
20    #[serde(default)]
21    pub description: String,
22    pub config: serde_json::Value,
23    #[serde(default)]
24    pub cases: Vec<ConfigTestCase>,
25    #[serde(default)]
26    pub expect_error: bool,
27}
28
29/// A test case within a config fixture.
30#[derive(Debug, Deserialize)]
31pub struct ConfigTestCase {
32    pub name: String,
33    pub context: HashMap<String, String>,
34    pub expect: Option<String>,
35}
36
37impl ConfigFixture {
38    /// Parse a single config fixture from YAML.
39    pub fn from_yaml(yaml: &str) -> Result<Self, serde_yaml::Error> {
40        serde_yaml::from_str(yaml)
41    }
42
43    /// Parse multiple config fixtures from a YAML file with `---` separators.
44    pub fn from_yaml_multi(yaml: &str) -> Result<Vec<Self>, serde_yaml::Error> {
45        let mut fixtures = Vec::new();
46        for doc in serde_yaml::Deserializer::from_str(yaml) {
47            fixtures.push(Self::deserialize(doc)?);
48        }
49        Ok(fixtures)
50    }
51}
52
53impl ConfigTestCase {
54    /// Build a `TestContext` from this case's context map.
55    #[must_use]
56    pub fn build_context(&self) -> crate::TestContext {
57        let mut ctx = crate::TestContext::new();
58        for (k, v) in &self.context {
59            ctx = ctx.with(k.clone(), v.clone());
60        }
61        ctx
62    }
63}
